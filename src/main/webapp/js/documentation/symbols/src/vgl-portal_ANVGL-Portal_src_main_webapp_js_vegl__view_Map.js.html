1 Ext.define("anvgl.view.Map", {
  2     /** @lends view.Map */
  3     
  4     extend: "Ext.panel.Panel",
  5     alias: "view.Map",
  6 
  7     id: "center_region",
  8     html: "<div style='width:100%; height:100%' id='center_region-map'></div>",
  9     region: "center",
 10     margin: '100 0 0 0',
 11 
 12     /**
 13      * Reads in the configuration object to work out the map, the layer store and any default base layer
 14      * @constructs
 15      *
 16      * @example
 17      *  var viewMap = Ext.create("anvgl.view.Map", {
 18      *       map: map, 
 19      *       defaultBaseLayerName : defaultBaseLayerName,
 20      *       layerStore: layerStore
 21      *  });
 22      * 
 23      * @param {object} config
 24      */
 25     constructor: function(config) {
 26         this.map = config.map;
 27         this.defaultBaseLayerName = config.defaultBaseLayerName || "";
 28         this.map.layerStore = config.layerStore;
 29         
 30         this.callParent(arguments);
 31     },
 32 
 33     /**
 34      * Listens to 'afterrender' to hook the map to the div (#center_region-map) 
 35      * and invokes a function to set the default base layer map
 36      * @listens
 37      */
 38     listeners: {
 39         afterrender: function () {
 40             this.map.renderToContainer(Ext.get("center_region-map"),"center_region-map");
 41             
 42             if (this.defaultBaseLayerName.length > 0) {
 43                 this.setDefaultLayer(this.map, this.defaultBaseLayerName);
 44             }
 45         }
 46     },
 47 
 48     /**
 49      * Reads the name of the default base layer, iterates and compares it to the map layers
 50      * and sets the default
 51      * @function
 52      * @param {object} map
 53      * @param {string} defaultBaseLayerName
 54      */
 55     setDefaultLayer : function(map, defaultBaseLayerName) {
 56         Ext.each(map.map.layers, function(layer) {
 57             if (layer.name === defaultBaseLayerName) {
 58                 map.map.setBaseLayer(layer);
 59                 return false;
 60             }
 61         });
 62     }
 63 });
