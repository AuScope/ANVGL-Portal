1 Ext.application({
  2     /** @lends anvgl */
  3     
  4     // application name
  5     name : "anvgl",
  6     
  7     // ExtJS application folder
  8     // Used to hook the application name to a base folder for namespaces
  9     appFolder : "js/vegl",
 10     
 11     // models / stores
 12     stores : ["FeaturedLayers"],
 13     
 14     // views
 15     views : [
 16              "Header", 
 17              "FeaturedLayers", 
 18              "KnownLayers", 
 19              "Tabs", 
 20              "Map",
 21              "Footer"
 22      ],
 23     
 24     // dependencies register
 25     requires : ["anvgl.util.handleException"],
 26     
 27     /**
 28       * Application launch.
 29       * Here we build our GUI from existing components - this function should only be assembling the GUI
 30       * Any processing logic should be managed in dedicated classes - don't let this become a
 31       * monolithic 'do everything' function
 32       * 
 33       * @constructs 
 34       *  
 35       */
 36     launch : function() {
 37         // map
 38         var map = null;
 39         
 40         // default base layer name
 41         var defaultBaseLayerName = "Google Satellite";
 42         
 43         // instantiate exception handler
 44         var handleException  = new anvgl.util.handleException();
 45         
 46         // send these headers with every AJax request we make to ensure we use gzip for most of our requests (where available)*/
 47         Ext.Ajax.defaultHeaders = {
 48             'Accept-Encoding': 'gzip, deflate' 
 49         };
 50 
 51         // create our store for holding the set of layers that have been added to the map
 52         var layerStore = Ext.create('portal.layer.LayerStore', {});
 53 
 54         // we need something to handle the clicks on the map
 55         var queryTargetHandler = Ext.create('portal.layer.querier.QueryTargetHandler', {});
 56         
 57         // create our map implementations
 58         var mapCfg = {
 59             container : null,   /** We will be performing a delayed render of this map */
 60             layerStore : layerStore,
 61             allowDataSelection : true,
 62             listeners : {
 63                 query : function(mapWrapper, queryTargets) {
 64                     queryTargetHandler.handleQueryTargets(mapWrapper, queryTargets);
 65                 }
 66             }
 67         };
 68         
 69         var urlParams = Ext.Object.fromQueryString(window.location.search.substring(1));
 70         var isDebugMode = urlParams.debug;
 71         
 72         if (urlParams && urlParams.map && urlParams.map === 'googleMap') {
 73             map = Ext.create('portal.map.gmap.GoogleMap', mapCfg);
 74         } else {
 75             map = Ext.create('portal.map.openlayers.OpenLayersMap', mapCfg);
 76         }
 77 
 78         var layerFactory = Ext.create('portal.layer.LayerFactory', {
 79             map : map,
 80             formFactory : Ext.create('vegl.layer.filterer.VeglFormFactory', {map : map}),
 81             downloaderFactory : Ext.create('vegl.layer.VeglDownloaderFactory', {map: map}),
 82             querierFactory : Ext.create('vegl.layer.VeglQuerierFactory', {map: map}),
 83             rendererFactory : Ext.create('vegl.layer.VeglRendererFactory', {map: map})
 84         });
 85         
 86         // layers sorter, passed to the FeaturedLayers store as a config parameter
 87         var layersSorter = new Ext.util.Sorter({
 88             sorterFn: function(record1, record2) {
 89                 var order1 = (record1.data.order.length ? record1.data.order : record1.data.name);
 90                 var order2 = (record2.data.order.length ? record2.data.order : record2.data.name);
 91                 return order1 > order2 ? 1 : (order1 < order2 ? -1 : 0);
 92             },
 93             direction: 'ASC'
 94         });
 95         
 96         // layers grouper, passed to the FeaturedLayers store as a config parameter
 97         var layersGrouper = new Ext.util.Grouper({
 98             groupFn: function(item) {
 99                 return item.data.group;
100             },
101             sorterFn: function(record1, record2) {
102                 var order1 = (record1.data.order.length ? record1.data.order : record1.data.group);
103                 var order2 = (record2.data.order.length ? record2.data.order : record2.data.group);
104                 return order1 > order2 ? 1 : (order1 < order2 ? -1 : 0);
105             },
106             direction: 'ASC'
107         });
108         
109         // featured layers store
110         var featuredLayerStore = Ext.create("store.FeaturedLayers", {
111             layersGrouper : layersGrouper,
112             layersSorter : layersSorter
113         });
114 
115         // 'layers' wrapper
116         var featuredLayers = Ext.create("view.FeaturedLayers", {
117             title : "Featured",
118             tooltip : {
119                  anchor : 'top',
120                  title : 'Featured Layers',
121                  text : '<p1>This is where the portal groups data services with a common theme under a layer. This allows you to interact with multiple data providers using a common interface.</p><br><p>The underlying data services are discovered from a remote registry. If no services can be found for a layer, it will be disabled.</p1>',
122                  showDelay : 100,
123                  icon : 'portal-core/img/information.png',
124                  dismissDelay : 30000
125             },
126             map : map,
127             activelayerstore : layerStore,
128             store : featuredLayerStore,
129             layerFactory : layerFactory
130         });
131         
132         // tabs view
133         var viewTabs = Ext.create("view.Tabs");
134         viewTabs.add([featuredLayers]);
135 
136        // layers view
137         var viewLayers = Ext.create("view.KnownLayers");
138         viewLayers.add(viewTabs);
139         
140         // map view
141         var viewMap = Ext.create("anvgl.view.Map", {
142             map: map, 
143             defaultBaseLayerName : defaultBaseLayerName,
144             layerStore: layerStore
145         });
146         
147         // footer view
148         var viewFooter = Ext.create("view.Footer");
149         viewLayers.add(viewFooter);
150         
151         // application view
152         var app = Ext.create('Ext.container.Viewport', {
153             layout:'border',
154             items:[viewLayers, viewMap]
155         });
156         
157         // the subset button needs a handler for when the user draws a subset bbox on the map:
158         map.on('dataSelect', function(map, bbox, intersectedRecords) {
159           /** show a dialog allow users to confirm the selected data sources */
160           if (intersectedRecords.length > 0) {
161               Ext.create('Ext.Window', {
162                   width : 710,
163                   maxHeight : 400,
164                   title : 'Confirm which datasets you wish to select',
165                   modal : true,
166                   autoScroll : true,
167                   items : [{
168                       xtype : 'dataselectionpanel',
169                       region : bbox,
170                       itemId : 'dataselection-panel',
171                       cswRecords : intersectedRecords
172                   }],
173                   buttons : [{
174                       text : 'Capture Data',
175                       iconCls : 'add',
176                       align : 'right',
177                       scope : this,
178                       handler : function(btn) {
179                           var parentWindow = btn.findParentByType('window');
180                           var panel = parentWindow.getComponent('dataselection-panel');
181 
182                           panel.saveCurrentSelection(function(totalSelected, totalErrors) {
183                               if (totalSelected === 0) {
184                                   Ext.Msg.alert('No selection', 'You haven\'t selected any data to capture. Please select one or more rows by checking the box alongside each row.');
185                               } else if (totalErrors === 0) {
186                                   Ext.Msg.alert('Request Saved', 'Your ' + totalSelected + ' dataset(s) have been saved. You can either continue selecting more data or <a href="jobbuilder.html">create a job</a> to process your existing selections.');
187                                   parentWindow.close();
188                               } else {
189                                   Ext.Msg.alert('Error saving data', 'There were one or more errors when saving some of the datasets you selected');
190                                   parentWindow.close();
191                               }
192                           });
193                       }
194                   }]
195               }).show();
196           }
197 
198         });
199 
200         // create our permalink generation handler
201         Ext.get('permanent-link').on('click', function() {
202             var mss = Ext.create('portal.util.permalink.MapStateSerializer');
203 
204             mss.addMapState(map);
205             mss.addLayers(layerStore);
206 
207             mss.serialize(function(state, version) {
208                 var popup = Ext.create('portal.widgets.window.PermanentLinkWindow', {
209                     state : state,
210                     version : version
211                 });
212 
213                 popup.show();
214             });
215         });
216         
217         // handle deserialisation -- ONLY if we have a uri param called "state"
218         var deserializationHandler;
219         var urlParams = Ext.Object.fromQueryString(window.location.search.substring(1));
220         if (urlParams && (urlParams.state || urlParams.s)) {
221             var decodedString = urlParams.state ? urlParams.state : urlParams.s;
222             var decodedVersion = urlParams.v;
223 
224             deserializationHandler = Ext.create('portal.util.permalink.DeserializationHandler', {
225                 knownLayerStore : knownLayerStore,
226                 layerFactory : layerFactory,
227                 layerStore : layerStore,
228                 map : map,
229                 stateString : decodedString,
230                 stateVersion : decodedVersion
231             });
232 
233         }
234     }
235 });
