1 /**
  2  * Configures the help messages depending on what VL page is or isn't loaded.
  3  *
  4  * This is seperated from the top level JS files so there is a physical distinction between
  5  * this help logic and the logic that runs the actual page.
  6  */
  7 Ext.define('vegl.HelpHandler', {
  8     statics : {
  9         manager : Ext.create('portal.util.help.InstructionManager', {}),
 10 
 11         /**
 12          * Asynchronously start an animation (with the specified delay in ms) that will highlight
 13          * the specified element with an animated arrow
 14          *
 15          * @param delay The delay in milli seconds
 16          * @param element An Ext.Element to highlight
 17          */
 18         highlightElement : function(delay, element) {
 19             var task = new Ext.util.DelayedTask(function() {
 20                 var arrowEl = Ext.getBody().createChild({
 21                     tag : 'img',
 22                     src : 'img/right-arrow.png',
 23                     width : '32',
 24                     height : '32',
 25                     style : {
 26                         'z-index' : 999999
 27                     }
 28                 });
 29 
 30                 //Figure out the x location of the element (in absolute page coords)
 31                 var xLocation = element.getLeft();
 32 
 33                 Ext.create('Ext.fx.Animator', {
 34                     target: arrowEl,
 35                     duration: 7000,
 36                     keyframes: {
 37                         0: {
 38                             opacity: 1
 39                         },
 40                         20: {
 41                             x: xLocation - 32
 42                         },
 43                         30: {
 44                             x: xLocation - 52
 45                         },
 46                         40: {
 47                             x: xLocation - 32
 48                         },
 49                         50: {
 50                             x: xLocation - 52
 51                         },
 52                         60: {
 53                             x: xLocation - 32
 54                         },
 55                         120: {
 56 
 57                         },
 58                         160: {
 59                             opacity : 0
 60                         }
 61                     },
 62                     listeners : {
 63                         afteranimate : Ext.bind(function(arrowEl) {
 64                             arrowEl.destroy();
 65                         }, this, [arrowEl])
 66                     }
 67                 });
 68             });
 69 
 70             task.delay(delay);
 71         }
 72     }
 73 
 74 }, function() {
 75     Ext.onReady(function() {
 76         var helpButtonEl = Ext.get('help-button');
 77 
 78         //If its a new user - highlight the help button an in unobtrusive manner
 79         if (window['NEW_SESSION'] === 'true') {
 80             vegl.HelpHandler.highlightElement(5000, helpButtonEl);
 81         }
 82 
 83         //Load help for main page
 84         if (window.location.pathname.endsWith('/gmap.html')) {
 85             helpButtonEl.on('click', function() {
 86                 vegl.HelpHandler.manager.showInstructions([Ext.create('portal.util.help.Instruction', {
 87                     highlightEl : 'auscope-tabs-panel',
 88                     title : 'Find data/layers',
 89                     description : 'In this panel a list of all available datasets in the form of layers will be presented to you.  Select the layer you would like to visualise.  Selecting a layer will also bring up any advanced filter options in the window below. <br/><br/>Further information about the data behind each layer can be displayed by clicking the icons alongside the layer name.'
 90                 }),Ext.create('portal.util.help.Instruction', {
 91                     highlightEl : 'center_region',
 92                     anchor : 'right',
 93                     title : 'Visualise Data',
 94                     description : 'The map panel here is where all of the currently added layers will be visualised. You can pan and zoom the map to an area of interest if required.'
 95                 }),Ext.create('portal.util.help.Instruction', {
 96                     highlightEl : 'gmap-subset-control',
 97                     anchor : 'right',
 98                     title : 'Select Data',
 99                     description : 'After reviewing one or more layers you can draw a region of interest using this button. All layers with data in the region you draw will be selected for use in a processing job. If the layer supports it, the data will be constrained to the region you select'
100                 }),Ext.create('portal.util.help.Instruction', {
101                     highlightEl : 'help-button',
102                     anchor : 'bottom',
103                     title : 'More information',
104                     description : 'For futher information, please consult the online <a target="_blank" href="https://www.seegrid.csiro.au/wiki/NeCTARProjects/VglUserGuide">VGL wiki</a>.'
105                 })]);
106             });
107         } else if (window.location.pathname.endsWith('/jobbuilder.html')) {
108             helpButtonEl.on('click', function() {
109                 var panel = Ext.getCmp('job-wizard-panel');
110                 var instructions = panel.getLayout().activeItem.items.get(0).getHelpInstructions();
111                 if (instructions && instructions.length) {
112                     vegl.HelpHandler.manager.showInstructions(instructions);
113                 } else {
114                     Ext.MessageBox.alert('No help', 'The current form doesn\'t have any help. For futher information, please consult the online <a target="_blank" href="https://www.seegrid.csiro.au/wiki/NeCTARProjects/VglUserGuide">VGL wiki</a>.');
115                 }
116             });
117         } else if (window.location.pathname.endsWith('/joblist.html')) {
118             helpButtonEl.on('click', function() {
119                 var vp = Ext.getCmp('vgl-joblist-viewport');
120                 var seriesP = vp.queryById('vgl-series-panel');
121                 var jobsP = vp.queryById('vgl-jobs-panel');
122                 var detailsP = vp.queryById('vgl-details-panel');
123                 var registerB = vp.queryById('btnRegister');
124 
125                 vegl.HelpHandler.manager.showInstructions([Ext.create('portal.util.help.Instruction', {
126                     highlightEl : seriesP.getEl(),
127                     title : 'Choose a Series',
128                     anchor : 'right',
129                     description : 'This panel will display every series that you\'ve created. Selecting a series will display all jobs owned by that series. You can manage an entire series by right clicking it or by selecting the desired series and pressing \'Actions\''
130                 }),Ext.create('portal.util.help.Instruction', {
131                     highlightEl : jobsP.getEl(),
132                     title : 'Manage Jobs',
133                     anchor : 'right',
134                     description : 'After selecting a series, a list of all jobs belonging to that series will be displayed here. Selecting a job will bring up information about that job in the details panel. You can manage individual jobs by right clicking them or by selecting a job and pressing \'Actions\'<br/><br/>If you\'d like more information about the job states, please consult the <a target="_blank" href="https://www.seegrid.csiro.au/wiki/VirtualLabs/VGL/VglUserGuide">VGL wiki</a>.'
135                 }),Ext.create('portal.util.help.Instruction', {
136                     highlightEl : detailsP.getEl(),
137                     title : 'Inspect Job',
138                     anchor : 'left',
139                     description : 'When you select a job, information about the job\'s metadata, logs and input/output files will be displayed here.<br/><br/>The description tab will provide a rough overview of the job. It will also contain a link to a remote provenance record (if one has been published).<br/><br/>The logs tab will display the entire console logs of a finished job. The logs will be divided into relevant sections. Please note that logs will not be available until after the job has finished processing.<br/><br/>The files tab contains all input/output files. Files from this list can be downloaded individually or as an archived zip file.'
140                 }),Ext.create('portal.util.help.Instruction', {
141                     highlightEl : registerB.getEl(),
142                     title : 'Register Job',
143                     anchor : 'top',
144                     description : 'If a job contains information that you would like to publish, selecting this button on a completed job will publish all provenance information to a remote Geonetwork registry. A link to the published record will be displayed in the \'Details\' Panel, under the \'Description\' tab.<br/><br/>Please note that this button will disabled unless you select a job whose status is \'Done\''
145                 })]);
146             });
147         } else {
148             helpButtonEl.hide();
149         }
150     });
151 });
