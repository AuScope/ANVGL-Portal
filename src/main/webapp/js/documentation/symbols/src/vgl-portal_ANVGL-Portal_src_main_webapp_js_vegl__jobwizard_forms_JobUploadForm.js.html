1 /**
  2  * Job wizard form for handling uploads of custom user input files
  3  *
  4  * Author - Josh Vote
  5  */
  6 Ext.define('vegl.jobwizard.forms.JobUploadForm', {
  7     extend : 'vegl.jobwizard.forms.BaseJobWizardForm',
  8 
  9     uploadedFilesStore : null,
 10     fileGrid : null,
 11 
 12     /**
 13      * Creates a new JobUploadForm form configured to write/read to the specified global state
 14      */
 15     constructor: function(wizardState) {
 16         var jobUploadFrm = this;
 17 
 18         this.callParent([{
 19             wizardState : wizardState,
 20             bodyStyle: 'padding:10px;',
 21             fileUpload: true,
 22             header : false,
 23             buttons: [],
 24             listeners : {
 25                 jobWizardActive : Ext.bind(jobUploadFrm.updateFileList, jobUploadFrm)
 26             },
 27             items: [{
 28                 xtype : 'jobinputfilespanel',
 29                 itemId : 'files-panel',
 30                 currentJobId : wizardState.jobId,
 31                 title: 'Input files',
 32                 stripeRows: true,
 33                 anchor: '100% -20',
 34                 buttons : [{
 35                     text : 'Add Input',
 36                     iconCls : 'add',
 37                     itemId : 'add-button',
 38                     align : 'right',
 39                     handler : function() {
 40                         Ext.create('vegl.widgets.JobInputFileWindow', {
 41                             jobId : jobUploadFrm.wizardState.jobId,
 42                             width : 500,
 43                             height : 300,
 44                             modal : true,
 45                             listeners : {
 46                                 close : function() {
 47                                     jobUploadFrm.updateFileList();
 48                                 }
 49                             }
 50                         }).show();
 51                     }
 52                 }]
 53             }]
 54         }]);
 55     },
 56 
 57     //Refresh the server side file list
 58     updateFileList : function() {
 59         var filesPanel = this.getComponent('files-panel');
 60         filesPanel.currentJobId = this.wizardState.jobId;
 61         filesPanel.updateFileStore();
 62     },
 63 
 64     //We don't validate on this form
 65     beginValidation : function(callback) {
 66         callback(true);
 67     },
 68 
 69     getTitle : function() {
 70         return "Manage job input files...";
 71     },
 72 
 73     getHelpInstructions : function() {
 74         var filesPanel = this.getComponent('files-panel');
 75         var addButton = filesPanel.queryById('add-button');
 76 
 77         return [Ext.create('portal.util.help.Instruction', {
 78             highlightEl : filesPanel.getEl(),
 79             title : 'Review Job Inputs',
 80             anchor : 'top',
 81             description : 'Every job has a number of input files. These files may yet to have been downloaded from a remote web service, or they may have already been uploaded from your desktop. Remote files will not be downloaded until after the job begins execution.<br/><br/>You can interact with these input files by right clicking them or by selecting them and pressing the \'Actions\' button.'
 82         }), Ext.create('portal.util.help.Instruction', {
 83             highlightEl : addButton.getEl(),
 84             title : 'Add more Inputs',
 85             anchor : 'top',
 86             description : 'If you\'d like to add your own custom input files/web service downloads then press this button. You will be prompted to either upload a file from your local machine or to specify the public URL where the file can be accessed from.'
 87         })];
 88     }
 89 });
