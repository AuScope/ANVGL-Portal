1 /**
  2  * A Ext.grid.Panel specialisation for rendering the Jobs
  3  * available to the current user.
  4  *
  5  * Adds the following events
  6  * selectjob : function(vegl.widgets.SeriesPanel panel, vegl.models.Job selection) - fires whenever a new Job is selected
  7  */
  8 Ext.define('vegl.widgets.JobFilesPanel', {
  9     extend : 'Ext.grid.Panel',
 10     alias : 'widget.jobfilespanel',
 11 
 12     currentJob : null,
 13     downloadAction : null,
 14     downloadZipAction : null,
 15 
 16     constructor : function(config) {
 17         var jobFilesGrid = this;
 18 
 19         //Action for downloading a single file
 20         this.downloadAction = new Ext.Action({
 21             text: 'Download',
 22             disabled: true,
 23             iconCls: 'disk-icon',
 24             handler: function() {
 25                 var fileRecord = jobFilesGrid.getSelectionModel().getSelection()[0];
 26 
 27                 var params = {
 28                     jobId : jobFilesGrid.currentJob.get('id'),
 29                     filename : fileRecord.get('name'),
 30                     key : fileRecord.get('name')
 31                 };
 32 
 33                 portal.util.FileDownloader.downloadFile("secure/downloadFile.do", params);
 34             }
 35         });
 36 
 37         //Action for downloading one or more files in a zip
 38         this.downloadZipAction = new Ext.Action({
 39             text: 'Download as Zip',
 40             disabled: true,
 41             iconCls: 'disk-icon',
 42             handler: function() {
 43                 var files = jobFilesGrid.getSelectionModel().getSelection();
 44 
 45                 var fParam = files[0].get('name');
 46                 for (var i = 1; i < files.length; i++) {
 47                     fParam += ',' + files[i].get('name');
 48                 }
 49 
 50                 portal.util.FileDownloader.downloadFile("secure/downloadAsZip.do", {
 51                     jobId : jobFilesGrid.currentJob.get('id'),
 52                     files : fParam
 53                 });
 54             }
 55         });
 56 
 57         Ext.apply(config, {
 58             plugins : [{
 59                 ptype : 'inlinecontextmenu',
 60                 align : 'left',
 61                 actions : [this.downloadAction]
 62             },{
 63                 ptype : 'rowcontextmenu',
 64                 contextMenu : Ext.create('Ext.menu.Menu', {
 65                     items: [this.downloadAction, this.downloadZipAction]
 66                 })
 67             }],
 68             multiSelect : true,
 69             store : Ext.create('Ext.data.Store', {
 70                 model : 'vegl.models.FileRecord',
 71                 proxy : {
 72                     type : 'ajax',
 73                     url : 'secure/jobFiles.do',
 74                     reader : {
 75                         type : 'json',
 76                         rootProperty : 'data'
 77                     },
 78                     listeners : {
 79                         exception : function(proxy, response, operation) {
 80                             responseObj = Ext.JSON.decode(response.responseText);
 81                             errorMsg = responseObj.msg;
 82                             errorInfo = responseObj.debugInfo;
 83                             portal.widgets.window.ErrorWindow.showText('Error', errorMsg, errorInfo);
 84                         }
 85                     }
 86                 }
 87             }),
 88             columns: [{ header: 'Filename', width: 200, sortable: true, dataIndex: 'name'},
 89                       { header: 'Size', width: 100, sortable: true, dataIndex: 'size', renderer: Ext.util.Format.fileSize, align: 'right'}],
 90             tbar: [{
 91                 text: 'Actions',
 92                 iconCls: 'folder-icon',
 93                 menu: [ this.downloadAction, this.downloadZipAction]
 94             }]
 95         });
 96 
 97         this.callParent(arguments);
 98 
 99         this.on('selectionchange', this._onSelectionChange, this);
100         this.on('celldblclick', this._onDblClick, this);
101 
102     },
103 
104     _onDblClick : function(view, td, cellIndex, record, tr, rowIndex, e, eOpts) {
105         var sm = this.getSelectionModel();
106 
107         this.getSelectionModel().select([record], false);
108         this.downloadAction.execute();
109     },
110 
111     _onSelectionChange : function(sm) {
112         var totalSelections = this.getSelectionModel().getSelection().length;
113         if (totalSelections === 0) {
114             this.downloadAction.setDisabled(true);
115             this.downloadZipAction.setDisabled(true);
116         } else {
117             if (totalSelections != 1) {
118                 this.downloadAction.setDisabled(true);
119             } else {
120                 this.downloadAction.setDisabled(false);
121             }
122             this.downloadZipAction.setDisabled(false);
123         }
124     },
125 
126     /**
127      * Reloads this store with all the jobs for the specified series
128      */
129     listFilesForJob : function(job) {
130         var store = this.getStore();
131         var ajaxProxy = store.getProxy();
132         ajaxProxy.extraParams.jobId = job.get('id');
133         this.currentJob = job;
134         store.removeAll(false);
135         store.load();
136     },
137 
138     /**
139      * Removes all files from the store and refresh the job files panel
140      */
141     cleanupDataStore : function() {
142         var store = this.getStore();
143         store.removeAll(false);
144     }
145 });
