1 /**
  2  * A class for creating instances of portal.layer.querier.Querier
  3  */
  4 Ext.define('vegl.layer.VeglQuerierFactory', {
  5 
  6     extend : 'portal.layer.querier.QuerierFactory',
  7 
  8 
  9     /**
 10      * Creates a new instance of a Querier based on the specified values
 11      *
 12      * knownLayer can be null
 13      */
 14     _generateQuerier : function(knownLayer, wfsResources, wmsResources, wcsResources) {
 15         var cfg = {map : this.map};
 16 
 17         //Geodesy features don't allow gml:Id lookups </rant>
 18         //To workaround this we have a custom feature source that looks up via the gps site id.
 19         if (knownLayer && knownLayer.get('id') === 'geodesy:gnssstation') {
 20             cfg.featureSource = Ext.create('portal.layer.querier.wfs.featuresources.WFSFeatureByPropertySource', {
 21                 property : 'GPSSITEID'
 22             });
 23         }
 24 
 25         if (knownLayer) {
 26             if (knownLayer.get('id') === 'gocad-models') {
 27                 return Ext.create('vegl.layer.querier.csw.SelectableCSWQuerier', cfg);
 28             } else if (knownLayer.get('id') === 'geophysics-datasets') {
 29                 return Ext.create('vegl.layer.querier.csw.GAProjectsQuerier', cfg);
 30             }
 31         }
 32 
 33         if (wfsResources.length > 0 && wmsResources.length > 0) {
 34             cfg.parser = Ext.create('vegl.layer.querier.wfs.VeglParser', {});
 35             cfg.knownLayerParser = Ext.create('vegl.layer.querier.wfs.VeglKnownLayerParser', {});
 36             return Ext.create('portal.layer.querier.wfs.WFSWithMapQuerier', cfg);
 37         } else if (wfsResources.length > 0) {
 38             cfg.parser = Ext.create('vegl.layer.querier.wfs.VeglParser', {});
 39             cfg.knownLayerParser = Ext.create('vegl.layer.querier.wfs.VeglKnownLayerParser', {});
 40             return Ext.create('portal.layer.querier.wfs.WFSQuerier', cfg);
 41         } else if (wcsResources.length > 0) {
 42             return Ext.create('portal.layer.querier.coverage.WCSQuerier',cfg);
 43         } else if (wmsResources.length > 0) {
 44             //Or just the plain old WMS querier
 45             return Ext.create('portal.layer.querier.wms.WMSQuerier', cfg);
 46         } else {
 47             //Worst case scenario, we render the source CSW record
 48             return Ext.create('portal.layer.querier.csw.CSWQuerier', cfg);
 49         }
 50     },
 51 
 52     /**
 53      * See parent class for defn
 54      */
 55     buildFromKnownLayer : function(knownLayer) {
 56         var allOnlineResources = knownLayer.getAllOnlineResources();
 57 
 58         var wmsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WMS);
 59         var wfsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WFS);
 60         var wcsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WCS);
 61 
 62         return this._generateQuerier(knownLayer, wfsResources, wmsResources, wcsResources);
 63     },
 64 
 65     /**
 66      * See parent class for defn
 67      */
 68     buildFromCswRecord : function(cswRecord) {
 69         var allOnlineResources = cswRecord.get('onlineResources');
 70 
 71         var wmsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WMS);
 72         var wfsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WFS);
 73         var wcsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WCS);
 74 
 75         return this._generateQuerier(null, wfsResources, wmsResources, wcsResources);
 76     }
 77 });
