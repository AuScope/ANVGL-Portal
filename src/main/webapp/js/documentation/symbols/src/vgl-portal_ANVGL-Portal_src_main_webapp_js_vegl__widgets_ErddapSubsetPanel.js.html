1 /**
  2  * A panel for allowing the user to confirm a sub selection of an ERDDAP dataset
  3  *
  4  */
  5 Ext.define('vegl.widgets.ErddapSubsetPanel', {
  6     extend : 'Ext.form.Panel',
  7 
  8     alias : 'widget.erddapsubsetpanel',
  9 
 10     /**
 11      * Accepts the following:
 12      * {
 13      *  region : portal.util.BBox - The selected area (defaults to 0,0,0,0)
 14      *  localPath : String - path to where the subset will be downloaded (on the VM) (defaults to '/tmp/subset-request')
 15      *  name : String - a descriptive name for this subset (defaults to 'ERDDAP Subset Request')
 16      *  description : String - a longer description for this subset (defaults to ''),
 17      *  coverageName : String - The name of the coverage (WCS name) being subsetted
 18      *  coverageUrl : String - the URL of the WCS which can be queried for more specific info on the coverage
 19      *  dataType : String - The default data type selection (defaults to 'nc' for NetCDF)
 20      * }
 21      */
 22     constructor : function(config) {
 23         var dataType = config.dataType ? config.dataType : 'nc';
 24         var region = config.region ? config.region : Ext.create('portal.util.BBox', {northBoundLatitude : 0,
 25             southBoundLatitude : 0,
 26             eastBoundLongitude : 0,
 27             westBoundLongitude : 0});
 28         var name = config.name ? config.name : 'ERDDAP Subset Request';
 29         var description = config.description ? config.description : '';
 30         var localPath = config.localPath ? config.localPath : '/tmp/subset-request';
 31 
 32         this.coverageName = config.coverageName;
 33         this.coverageUrl = config.coverageUrl;
 34 
 35         // subset file type selection values
 36         var fileTypeStore = Ext.create('Ext.data.Store', {
 37             fields : [{name : 'urn'}, {name : 'label'}],
 38             data : [{label : 'CSV', urn : 'csv'},
 39                     {label : 'GeoTIFF', urn : 'geotif'},
 40                     {label : 'NetCDF', urn : 'nc'}]
 41         });
 42 
 43 
 44         Ext.apply(config, {
 45             items : [{
 46                 xtype : 'combo',
 47                 anchor : '100%',
 48                 editable : false,
 49                 forceSelection : true,
 50                 fieldLabel : 'Data Type',
 51                 mode : 'local',
 52                 store : fileTypeStore,
 53                 triggerAction : 'all',
 54                 typeAhead : false,
 55                 displayField : 'label',
 56                 valueField : 'urn',
 57                 value : dataType,
 58                 name : 'format',
 59                 plugins: [{
 60                     ptype: 'fieldhelptext',
 61                     text: 'The data format to request from the remote service.'
 62                 }]
 63             },{
 64                 xtype: 'fieldcontainer',
 65                 fieldLabel: 'Region',
 66                 msgTarget: 'under',
 67                 anchor: '100%',
 68                 layout: {
 69                     type: 'hbox',
 70                     defaultMargins: {top: 0, right: 5, bottom: 0, left: 0}
 71                 },
 72                 defaults: {
 73                     hideLabel: true,
 74                     decimalPrecision : 8
 75                 },
 76                 items: [
 77                     {xtype: 'displayfield', value: '('},
 78                     {xtype: 'numberfield',  name: 'northBoundLatitude', width: 80, value: region.northBoundLatitude, listeners : {change : Ext.bind(this.updateSelectionSize, this)}},
 79                     {xtype: 'displayfield', value: 'North ) ('},
 80                     {xtype: 'numberfield',  name: 'westBoundLongitude', width: 80, value: region.westBoundLongitude, listeners : {change : Ext.bind(this.updateSelectionSize, this)}},
 81                     {xtype: 'displayfield', value: 'West ) ('},
 82                     {xtype: 'numberfield',  name: 'southBoundLatitude', width: 80, value: region.southBoundLatitude, listeners : {change : Ext.bind(this.updateSelectionSize, this)}},
 83                     {xtype: 'displayfield', value: 'South ) ('},
 84                     {xtype: 'numberfield',  name: 'eastBoundLongitude', width: 80, value : region.eastBoundLongitude, listeners : {change : Ext.bind(this.updateSelectionSize, this)}},
 85                     {xtype: 'displayfield', value: 'East )'}
 86                 ],
 87                 plugins: [{
 88                     ptype: 'fieldhelptext',
 89                     text: 'The spatial region subset to request from the data service. The values represent a WGS:84 bounding box.'
 90                 }]
 91             },{
 92                 xtype : 'displayfield',
 93                 fieldLabel : 'Selection Size',
 94                 itemId : 'selection-size',
 95                 anchor : '100%',
 96                 plugins: [{
 97                     ptype: 'fieldhelptext',
 98                     text: 'This selection size is only an approximation. It should only be used as a rough gauge of how much data you are requesting.'
 99                 }]
100             },{
101                 xtype : 'textfield',
102                 fieldLabel : 'Location',
103                 value : localPath,
104                 anchor : '100%',
105                 name : 'localPath',
106                 plugins: [{
107                     ptype: 'fieldhelptext',
108                     text: 'A file path where requested data will be downloaded to. It will be accessible from within your job script.'
109                 }]
110             },{
111                 xtype : 'textfield',
112                 fieldLabel : 'Name',
113                 value : name,
114                 anchor : '100%',
115                 name : 'name',
116                 plugins: [{
117                     ptype: 'fieldhelptext',
118                     text: 'A short name to describe this download request.'
119                 }]
120             },{
121                 xtype : 'textarea',
122                 fieldLabel : 'Description',
123                 value : description,
124                 anchor : '100%',
125                 name : 'description',
126                 plugins: [{
127                     ptype: 'fieldhelptext',
128                     text: 'A longer description to outline the purpose of this download request.'
129                 }]
130             }]
131         });
132 
133         this.callParent(arguments);
134 
135         this.on('render', this.updateSelectionSize, this);
136     },
137 
138     updateSelectionSize : function() {
139         var ssField = this.getComponent('selection-size');
140         var values = this.getForm().getValues();
141 
142         ssField.setValue('Loading...');
143 
144         vegl.util.WCSUtil.estimateCoverageSize(values, this.coverageUrl, this.coverageName, ssField, function(success, msg, response, ssField) {
145             if (!success) {
146                 ssField.setValue(msg);
147                 return;
148             }
149 
150             var approxTotal = Ext.util.Format.number(vegl.util.WCSUtil.roundToApproximation(response.roundedTotal), '0,000');
151             var approxSize = Ext.util.Format.fileSize(vegl.util.WCSUtil.roundToApproximation(response.width * response.height * 4));
152             ssField.setValue(Ext.util.Format.format('Approximately <b>{0}</b> data points in total. Uncompressed that\'s roughly {1}', approxTotal, approxSize));
153         });
154     }
155 });
