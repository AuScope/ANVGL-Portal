1 /**
  2  * Class for parsing a set of portal.csw.CSWRecord objects request/response
  3  * using the Querier interface
  4  *
  5  * The resulting Querier allows the selection of the CSWRecord online resources
  6  * such that they will be downloaded by a VL job.
  7  */
  8 Ext.define('vegl.layer.querier.csw.SelectableCSWQuerier', {
  9     extend: 'portal.layer.querier.Querier',
 10 
 11     constructor: function(config){
 12         this.callParent(arguments);
 13     },
 14 
 15     /**
 16      * See parent class for definition
 17      */
 18     query : function(queryTarget, callback) {        
 19         var cswRecord = queryTarget.get('cswRecord');
 20         if (!cswRecord) {
 21             callback(this, [], queryTarget);
 22             return;
 23         }
 24 
 25         var keywordsString = "";
 26         var keywords = cswRecord.get('descriptiveKeywords');
 27         for (var i = 0; i < keywords.length; i++) {
 28             keywordsString += keywords[i];
 29             if (i < (keywords.length - 1)) {
 30                 keywordsString += ', ';
 31             }
 32         }
 33 
 34         var panel = Ext.create('portal.layer.querier.BaseComponent', {
 35             border : false,
 36             autoScroll : true,
 37             items : [{
 38                 layout : 'fit',
 39                 items : [{
 40                     xtype : 'fieldset',
 41                     items : [{
 42                         xtype : 'displayfield',
 43                         fieldLabel : 'Source',
 44                         value : Ext.util.Format.format('<a target="_blank" href="{0}">Link back to registry</a>', cswRecord.get('recordInfoUrl'))
 45                     },{
 46                         xtype : 'displayfield',
 47                         fieldLabel : 'Title',
 48                         anchor : '100%',
 49                         value : cswRecord.get('name')
 50                     }, {
 51                         xtype : 'textarea',
 52                         fieldLabel : 'Abstract',
 53                         anchor : '100%',
 54                         value : cswRecord.get('description'),
 55                         readOnly : true
 56                     },{
 57                         xtype : 'displayfield',
 58                         fieldLabel : 'Keywords',
 59                         anchor : '100%',
 60                         value : keywordsString
 61                     },{
 62                         xtype : 'displayfield',
 63                         fieldLabel : 'Contact Org',
 64                         anchor : '100%',
 65                         value : cswRecord.get('contactOrg')
 66                     },{
 67                         xtype : 'dataselectionpanel',
 68                         itemId : 'dataselection-panel',
 69                         region : cswRecord.get('geographicElements'),
 70                         hideHeaders : false,
 71                         cswRecords : [ cswRecord ]
 72                     },{
 73                         xtype : 'button',
 74                         margin : '5 0 0 0',
 75                         text : 'Capture selected',
 76                         iconCls : 'add',
 77                         handler : Ext.bind(this.addSelectedResourcesToSession, this) 
 78                     }]
 79                 }]
 80             }]
 81         });
 82 
 83         callback(this, [panel], queryTarget);
 84     },
 85     
 86     addSelectedResourcesToSession : function(button) {
 87         var fieldSetContainer = button.findParentByType('fieldset');
 88         var dataSelPanel = fieldSetContainer.getComponent('dataselection-panel');
 89         
 90         dataSelPanel.saveCurrentSelection(function(totalSelected, totalErrors) {
 91             if (totalSelected === 0) {
 92                 Ext.Msg.alert('No selection', 'You haven\'t selected any data to capture. Please select one or more rows by checking the box alongside each row.');
 93             } else if (totalErrors === 0) {
 94                 Ext.Msg.alert('Request Saved', 'Your ' + totalSelected + ' dataset(s) have been saved. You can either continue selecting more data or <a href="jobbuilder.html">create a job</a> to process your existing selections.');
 95                 this.close();
 96             } else {
 97                 Ext.Msg.alert('Error saving data', 'There were one or more errors when saving some of the datasets you selected');
 98                 this.close();
 99             }
100         });
101     }
102 });
