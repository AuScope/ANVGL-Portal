1 /**
  2  * A panel for allowing the user to confirm a sub selection of an NetCdf dataset
  3  *
  4  */
  5 Ext.define('vegl.widgets.NetcdfSubsetPanel', {
  6     extend : 'Ext.form.Panel',
  7 
  8     alias : 'widget.netcdfsubsetpanel',
  9 
 10     /**
 11      * Accepts the following:
 12      * {
 13      *  region : portal.util.BBox - The selected area (defaults to 0,0,0,0)
 14      *  localPath : String - path to where the subset will be downloaded (on the VM) (defaults to '/tmp/subset-request')
 15      *  name : String - a descriptive name for this subset (defaults to 'Netcdf Subset Request')
 16      *  description : String - a longer description for this subset (defaults to ''),
 17      *  coverageName : String - The name of the coverage (WCS name) being subsetted
 18      *  coverageUrl : String - the URL of the WCS which can be queried for more specific info on the coverage
 19      *  dataType : String - The default data type selection (defaults to 'nc' for NetCDF)
 20      * }
 21      */
 22     constructor : function(config) {
 23         var dataType = config.dataType ? config.dataType : 'nc';
 24         var region = config.region ? config.region : Ext.create('portal.util.BBox', {northBoundLatitude : 0,
 25             southBoundLatitude : 0,
 26             eastBoundLongitude : 0,
 27             westBoundLongitude : 0});
 28         var description = config.description ? config.description : '';
 29         var localPath = config.localPath ? config.localPath : '/tmp/subset-request';
 30 
 31         this.coverageName = config.coverageName;
 32         this.coverageUrl = config.coverageUrl;
 33 
 34         // subset file type selection values
 35         var fileTypeStore = Ext.create('Ext.data.Store', {
 36             fields : [{name : 'urn'}, {name : 'label'}],
 37             data : [{label : 'CSV', urn : 'csv'},
 38                     {label : 'GeoTIFF', urn : 'geotif'},
 39                     {label : 'NetCDF', urn : 'nc'}]
 40         });
 41 
 42 
 43         Ext.apply(config, {
 44             items : [{
 45                 xtype: 'fieldcontainer',
 46                 fieldLabel: 'Region',
 47                 msgTarget: 'under',
 48                 anchor: '100%',
 49                 layout: {
 50                     type: 'hbox',
 51                     defaultMargins: {top: 0, right: 5, bottom: 0, left: 0}
 52                 },
 53                 defaults: {
 54                     hideLabel: true,
 55                     decimalPrecision : 8
 56                 },
 57                 items: [
 58                     {xtype: 'displayfield', value: '('},
 59                     {xtype: 'numberfield',  name: 'northBoundLatitude', width: 80, value: region.northBoundLatitude, listeners : {change : Ext.bind(this.updateSelectionSize, this)}},
 60                     {xtype: 'displayfield', value: 'North ) ('},
 61                     {xtype: 'numberfield',  name: 'westBoundLongitude', width: 80, value: region.westBoundLongitude, listeners : {change : Ext.bind(this.updateSelectionSize, this)}},
 62                     {xtype: 'displayfield', value: 'West ) ('},
 63                     {xtype: 'numberfield',  name: 'southBoundLatitude', width: 80, value: region.southBoundLatitude, listeners : {change : Ext.bind(this.updateSelectionSize, this)}},
 64                     {xtype: 'displayfield', value: 'South ) ('},
 65                     {xtype: 'numberfield',  name: 'eastBoundLongitude', width: 80, value : region.eastBoundLongitude, listeners : {change : Ext.bind(this.updateSelectionSize, this)}},
 66                     {xtype: 'displayfield', value: 'East )'}
 67                 ],
 68                 plugins: [{
 69                     ptype: 'fieldhelptext',
 70                     text: 'The spatial region subset to request from the data service. The values represent a WGS:84 bounding box.'
 71                 }]
 72             },{
 73                 xtype : 'textfield',
 74                 fieldLabel : 'Location',
 75                 value : localPath,
 76                 anchor : '100%',
 77                 name : 'localPath',
 78                 plugins: [{
 79                     ptype: 'fieldhelptext',
 80                     text: 'A file path where requested data will be downloaded to. It will be accessible from within your job script.'
 81                 }]
 82             },{
 83                 xtype : 'textfield',
 84                 fieldLabel : 'Name',
 85                 value : name,
 86                 anchor : '100%',
 87                 name : 'name',
 88                 plugins: [{
 89                     ptype: 'fieldhelptext',
 90                     text: 'A short name to describe this download request.'
 91                 }]
 92             },{
 93                 xtype : 'textarea',
 94                 fieldLabel : 'Description',
 95                 value : description,
 96                 anchor : '100%',
 97                 name : 'description',
 98                 plugins: [{
 99                     ptype: 'fieldhelptext',
100                     text: 'A longer description to outline the purpose of this download request.'
101                 }]
102             }]
103         });
104 
105         this.callParent(arguments);
106     }
107 });
