1 /**
  2  * Class for parsing a set of portal.csw.CSWRecord objects request/response
  3  * using the Querier interface
  4  * 
  5  * The resulting Querier is specialised for rendering a parent/child
  6  * relationship of CSW records. The GUI is specialised for Geoscience Australia
  7  * ISO records for their gridded /elevation/magnetics/radiometrics data.
  8  */
  9 Ext.define('vegl.layer.querier.csw.GAProjectsQuerier', {
 10     extend: 'portal.layer.querier.Querier',
 11 
 12     constructor: function(config){
 13         this.callParent(arguments);
 14     },
 15 
 16     _getFirstCSWRecordWithKeyword : function(cswRecords, keywords) {
 17         for (var i = 0; i < cswRecords.length; i++) {
 18             if (cswRecords[i].containsKeywords(keywords)) {
 19                 return cswRecords[i];
 20             }
 21         }
 22         return null;
 23     },
 24     
 25     _getFirstMatchingGridInfoValue : function(griddedInfo, name, uom) {
 26         for (var i = 0; i < griddedInfo.length; i++) {
 27             if (griddedInfo[i].nameOfMeasure === name && 
 28                 griddedInfo[i].unitOfMeasure === uom) {
 29                 return griddedInfo[i].value;
 30             }
 31         }
 32         return null;
 33     },
 34     
 35     _createNamedTabForCSWRecord : function(title, cswRecord) {
 36         var griddedInfo = cswRecord.get('extensions').griddedInfo;
 37         var cellSizeM = this._getFirstMatchingGridInfoValue(griddedInfo, "CellSize", "metres");
 38         var cellSizeDD = this._getFirstMatchingGridInfoValue(griddedInfo, "CellSize", "decimal degrees");
 39         var lineSpacingM = this._getFirstMatchingGridInfoValue(griddedInfo, "LineSpacing", "metres");
 40         
 41         //Configure default download options
 42         var wcsResources = portal.csw.OnlineResource.getFilteredFromArray(cswRecord.get('onlineResources'), portal.csw.OnlineResource.WCS);
 43         var dlOptions = null;
 44         var wcsResource = null;
 45         if (wcsResources.length > 0) {
 46             var geoEls = cswRecord.get('geographicElements');
 47             var defaultBbox = (geoEls.length > 0) ? geoEls[0] : null;
 48             
 49             wcsResource = wcsResources[0];
 50             dlOptions = vegl.util.DataSelectionUtil.createDownloadOptionsForResource(wcsResource, cswRecord, defaultBbox);
 51         }
 52         
 53         //Build a preview URL
 54         var geoEls = cswRecord.get('geographicElements');
 55         var wmsResources = portal.csw.OnlineResource.getFilteredFromArray(cswRecord.get('onlineResources'), portal.csw.OnlineResource.WMS);
 56         var previewUrl = null;
 57         var previewWidth = 1;
 58         var previewHeight = 1;
 59         if (geoEls.length > 0 && wmsResources.length > 0) {
 60             var superBbox = geoEls[0];
 61             for (var i = 1; i < geoEls.length; i++) {
 62                 superBbox = superBbox.combine(geoEls[i]);
 63             }
 64     
 65             //Set our Height to a constant and scale the height appropriately
 66             var widthRatio = (superBbox.eastBoundLongitude - superBbox.westBoundLongitude) /
 67                              (superBbox.northBoundLatitude - superBbox.southBoundLatitude);
 68             var height = 128;
 69             var width = Math.floor(height * widthRatio);
 70     
 71             previewWidth = width;
 72             previewHeight = height;
 73     
 74             var url = wmsResources[0].get('url');
 75             var name = wmsResources[0].get('name');
 76             if (cswRecord.get('version')=='1.3.0') {
 77                 previewUrl = portal.map.primitives.BaseWMSPrimitive.getWms_130_Url(url, name, superBbox, width, height); 
 78             } else {
 79                 previewUrl = portal.map.primitives.BaseWMSPrimitive.getWmsUrl(url, name, superBbox, width, height);
 80             }
 81         }
 82         
 83         return {
 84             title : title,
 85             _dlOptions : dlOptions,
 86             _wcsResource : wcsResource,
 87             listeners : {
 88                 render : Ext.bind(function(container, eOpts, dlOptions, wcsResource) {
 89                     //Save our dlOptions to this container for our buttons to write/retrieve
 90                     container._dlOptions = dlOptions;
 91                     container._wcsResource = wcsResource;
 92                 }, this, [dlOptions, wcsResource], true)
 93             },
 94             items : [{
 95                 xtype : 'label',
 96                 style : 'font-size: 11px; font-style:normal;',
 97                 html : Ext.util.Format.format('<a target="_blank" href="{1}">{0}</a>', cswRecord.get('name'), cswRecord.get('recordInfoUrl')) 
 98             },{
 99                 layout : 'hbox',
100                 border : false,
101                 items : [{
102                     xtype : 'container',
103                     flex : 1,
104                     height : 100,
105                     layout : {
106                         type : 'hbox',
107                         pack : 'center',
108                         align : 'middle'
109                     },
110                     items : [{
111                         xtype : 'datadisplayfield',
112                         uom : 'm',
113                         uomTip : 'metres',
114                         fieldLabel : 'Cell Size',
115                         value : cellSizeM ? cellSizeM : 'Unknown',
116                     }]
117                 },{
118                     xtype : 'container',
119                     flex : 1,
120                     height : 100,
121                     layout : {
122                         type : 'hbox',
123                         pack : 'center',
124                         align : 'middle'
125                     },
126                     items : [{
127                         xtype : 'datadisplayfield',
128                         uom : 'DD',
129                         uomTip : 'decimal degrees',
130                         fieldLabel : 'Cell Size',
131                         value : cellSizeDD ? cellSizeDD : 'Unknown',
132                     }]
133                 },{
134                     xtype : 'container',
135                     flex : 1,
136                     height : 100,
137                     layout : {
138                         type : 'hbox',
139                         pack : 'center',
140                         align : 'middle'
141                     },
142                     items : [{
143                         xtype : 'datadisplayfield',
144                         uom : 'm',
145                         uomTip : 'metres',
146                         fieldLabel : 'Line Spacing',
147                         value : lineSpacingM ? lineSpacingM : 'Unknown',
148                     }]
149                 },{
150                     xtype : 'imagedisplayfield',
151                     flex : 1,
152                     imgHref : previewUrl,
153                     imgWidth : previewWidth,
154                     imgHeight : previewHeight,
155                     fieldLabel : "Grid Preview"
156                 }]
157             },{
158                 xtype : 'button',
159                 iconCls : 'add',
160                 text : 'Capture this grid',
161                 hidden : dlOptions === null,
162                 style : 'position: absolute; bottom: 10px; left: 5px',
163                 handler : function(btn) {
164                     var dlOptions = btn.ownerCt._dlOptions;
165                     var wcsResource = btn.ownerCt._wcsResource;
166                     
167                     var myMask = new Ext.LoadMask({
168                         msg: "Capturing grid...",
169                         target : this
170                             });
171                     myMask.show();
172                     
173                     vegl.util.DataSelectionUtil.makeDownloadUrl(wcsResource, dlOptions, true, function(success) {
174                         myMask.hide();
175                         if (success) {
176                             Ext.Msg.alert('Request Saved', 'Your dataset has been saved. You can either continue selecting more data or <a href="jobbuilder.html">create a job</a> to process your existing selections.');
177                         } else {
178                             Ext.Msg.alert('Error saving data', 'There were one or more errors when saving some of the datasets you selected');
179                         }
180                     });
181                 }
182             },{
183                 xtype : 'button',
184                 iconCls : 'edit',
185                 text : 'Edit this grid',
186                 hidden : dlOptions === null,
187                 style : 'position: absolute; bottom: 10px; left: 130px',
188                 handler : function(btn) {
189                     var dlOptions = btn.ownerCt._dlOptions;
190                     var wcsResource = btn.ownerCt._wcsResource;
191                     vegl.util.DataSelectionUtil.showDownloadOptionsForResource(wcsResource, dlOptions, Ext.bind(function(newDlOptions) {
192                         this.ownerCt._dlOptions = newDlOptions;
193                     }, btn));
194                 }
195                     
196             }]
197         };
198     },
199     
200     
201     /**
202      * See parent class for definition
203      */
204     query : function(queryTarget, callback) {
205         var cswRecord = queryTarget.get('cswRecord');
206         if (!cswRecord) {
207             callback(this, [], queryTarget);
208             return;
209         }
210         
211         //source url/domain
212         var sourceUrl = cswRecord.get('recordInfoUrl');
213         var urlRegexp = /^((http[s]?|ftp):\/)?\/?([^:\/\s]+)((\/\w+)*\/)([\w\-\.]+[^#?\s]+)(.*)?(#[\w\-]+)?$/;
214         var match = urlRegexp.exec(sourceUrl);
215         var sourceHost = match[3];
216         
217         var dateStamp = cswRecord.get('extensions').dateStamp;
218         if (dateStamp) {
219             dateStamp = '(' + dateStamp + ')';
220         } else {
221             dateStamp = '';
222         }
223             
224         //Create gridded info
225         var childRecords = cswRecord.get('childRecords');
226         var magRecord = this._getFirstCSWRecordWithKeyword(childRecords, ["MAG", "MAGNETICS"]);
227         var elevationRecord = this._getFirstCSWRecordWithKeyword(childRecords, ["ELE", "ELEVATION"]);
228         var radRecord = this._getFirstCSWRecordWithKeyword(childRecords, ["RAD", "RADIOMETRICS"]);
229         var tabs = [];
230         if (elevationRecord) {
231             tabs.push(this._createNamedTabForCSWRecord("Elevation Grid", elevationRecord));
232         }
233         if (magRecord) {
234             tabs.push(this._createNamedTabForCSWRecord("Magnetics Grid", magRecord));
235         }
236         if (radRecord) {
237             tabs.push(this._createNamedTabForCSWRecord("Radiometrics Grid", radRecord));
238         }
239         
240         var panel = Ext.create('portal.layer.querier.BaseComponent', {
241             border : false,
242             autoScroll : true,
243             items : [{
244                 layout : 'fit',
245                 border : false,
246                 items : [{
247                     xtype : 'fieldset',
248                     title : Ext.util.Format.format('<span style="font-size:16px;">{0} {1}</span>',cswRecord.get('name'), dateStamp),
249                     border : false,
250                     items : [{
251                         xtype : 'label',
252                         anchor : '100%',
253                         style : "font-size: 12px; font-style:italic;",
254                         html : cswRecord.get('description') + '<br/><br/>'
255                     },{
256                         xtype : 'label',
257                         anchor : '100%',
258                         style : "font-size: 12px;",
259                         html : Ext.util.Format.format('<span>This metadata is sourced from <a target="_blank" href="{1}">{0}</a></span>', sourceHost, sourceUrl)
260                     }, {
261                         xtype : 'tabpanel',
262                         padding : '20 0 0 0', 
263                         tabBar : {
264                             plain : true
265                         },
266                         items : tabs
267                     }]
268                 }]
269             }]
270         });
271 
272         callback(this, [panel], queryTarget);
273     }
274 });
