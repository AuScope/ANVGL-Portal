1 /**
  2  * A Ext.form.field.ComboBox specialisation for choosing from a selection of vegl.models.MachineImage
  3  */
  4 Ext.define('vegl.widgets.MachineImageCombo', {
  5     extend : 'Ext.form.field.ComboBox',
  6     alias : 'widget.machineimagecombo',
  7 
  8     /**
  9      * Accepts an Ext.field.Combo. It is expected that the internal store will be populated
 10      * using vegl.models.MachineImage objects.
 11      */
 12     constructor : function(config) {
 13         var listCfg = config.listConfig ? config.listConfig : {};
 14         listCfg.getInnerTpl = function() {
 15             // Custom rendering template for each item
 16             return '<h3 style="margin:0px;">{name}</h3>' +
 17                    '{description}<br/>' +
 18                    '<ul>' +
 19                        '<tpl for="keywords"><li style="list-style:disc inside none;">{.}</li></tpl>' +
 20                    '</ul>';
 21         };
 22 
 23         Ext.apply(config, {
 24             displayField: 'name',
 25             valueField : 'imageId',
 26             listConfig: listCfg
 27         });
 28 
 29         this.callParent(arguments);
 30     }
 31 });
