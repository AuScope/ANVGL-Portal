1 /**
  2  * A class for creating instances of portal.layer.renderer.Renderer
  3  */
  4 Ext.define('vegl.layer.VeglRendererFactory', {
  5 
  6     extend : 'portal.layer.renderer.RendererFactory',
  7 
  8     /**
  9      * Creates a new instance of renderer based on the specified values
 10      */
 11     _generateRenderer : function(source, wfsResources, wmsResources, proxyUrl, proxyCountUrl, iconUrl, iconSize, iconAnchor) {
 12         var icon = Ext.create('portal.map.Icon', {
 13             url : iconUrl,
 14             width : iconSize ? iconSize.width : 16,
 15             height : iconSize ? iconSize.height : 16,
 16             anchorOffsetX : iconAnchor ? iconAnchor.x : 0,
 17             anchorOffsetY : iconAnchor ? iconAnchor.y : 0
 18         });
 19 
 20         if (source instanceof portal.knownlayer.KnownLayer) {
 21             switch (source.get('id')) {
 22                 case 'geophysics-datasets':
 23                     return Ext.create('portal.layer.renderer.csw.CSWRenderer', {
 24                         map : this.map,
 25                         icon : icon
 26                     });
 27             }
 28         }
 29 
 30         if (wmsResources.length > 0) {
 31             return Ext.create('portal.layer.renderer.wms.LayerRenderer', {map : this.map});
 32         } else if (wfsResources.length > 0) {
 33             return Ext.create('portal.layer.renderer.wfs.FeatureRenderer', {
 34                 map : this.map,
 35                 icon : icon,
 36                 proxyUrl : proxyUrl ? proxyUrl : 'getAllFeatures.do',
 37                 proxyCountUrl : proxyCountUrl
 38             });
 39         }else{
 40             return Ext.create('portal.layer.renderer.csw.CSWRenderer', {
 41                 map : this.map,
 42                 icon : icon
 43             });
 44         }
 45     },
 46 
 47     /**
 48      * See parent class for defn
 49      */
 50     buildFromKnownLayer : function(knownLayer) {
 51         var allOnlineResources = knownLayer.getAllOnlineResources();
 52 
 53         var wmsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WMS);
 54         var wfsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WFS);
 55         var wcsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WCS);
 56 
 57         return this._generateRenderer(knownLayer, wfsResources, wmsResources, knownLayer.get('proxyUrl'), knownLayer.get('proxyCountUrl'),
 58                 knownLayer.get('iconUrl'), knownLayer.get('iconSize'), knownLayer.get('iconAnchor'));
 59     },
 60 
 61     /**
 62      * See parent class for defn
 63      */
 64     buildFromCswRecord : function(cswRecord) {
 65         var allOnlineResources = cswRecord.get('onlineResources');
 66 
 67         var wmsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WMS);
 68         var wfsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WFS);
 69         var wcsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WCS);
 70 
 71         return this._generateRenderer(cswRecord, wfsResources, wmsResources, undefined, undefined, undefined, undefined, undefined);
 72     }
 73 });
