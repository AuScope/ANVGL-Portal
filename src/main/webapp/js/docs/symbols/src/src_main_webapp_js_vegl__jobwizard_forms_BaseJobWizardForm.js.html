1 /**
  2 * The Base object for all Job wizard steps to inherit from
  3 *
  4 * Author - Josh Vote
  5 */
  6 Ext.namespace("JobBuilder");
  7 Ext.require([
  8              'Ext.ux.form.plugin.FieldHelpText'
  9  ]);
 10 Ext.define('vegl.jobwizard.forms.BaseJobWizardForm', {
 11     extend : 'Ext.form.Panel',
 12 
 13 
 14 
 15     /**
 16      * State object that is shared by all wizard forms. Use it to communicate
 17      * work state between steps (although ideally a job's dependency on this object
 18      * should be kept to a bare minimum)
 19      */
 20     wizardState : {},
 21 
 22     /**
 23      * Set this to true in a child class to skip the warning when the user attempts to navigate
 24      * away from the page
 25      */
 26     noWindowUnloadWarning : false,
 27 
 28     /**
 29      * An array of button objects/config objects that will be inserted between the previous and next buttons.
 30      */
 31     additionalButtons : null,
 32 
 33     constructor: function(obj) {
 34         if (!obj.style) {
 35             obj.style = {};
 36         }
 37         obj.style['background-color'] = 'white';
 38         obj.border = false;
 39 
 40         this.callParent(arguments);
 41 
 42         this.wizardState = obj.wizardState;
 43 
 44         this.on('jobWizardActive', function() {
 45             window.addEventListener('beforeunload',this.onBeforeWindowUnload,this)
 46         }, this);
 47 
 48         this.on('jobWizardDeactive', function() {
 49             window.removeEventListener('beforeunload',this.onBeforeWindowUnload,this)
 50         }, this);
 51     },
 52 
 53     /**
 54      * Handler that is called whenever the user attempts to navigate away from the page whilst this
 55      * job configuration step is active.
 56      */
 57     onBeforeWindowUnload : function(e) {
 58         if (!this.noWindowUnloadWarning) {
 59             e.browserEvent.returnValue = "All entered details will be lost!";
 60         }
 61     },
 62 
 63     /**
 64      * [abstract] This function should begin the validation of this wizard step asynchronously.
 65      * and the result of the validation should be passed back to the callback function.
 66      *
 67      * If the validation result is false then the user will not be permitted to progress
 68      * the workflow
 69      *
 70      * callback : function(success)
 71      */
 72     beginValidation : function(callback) {
 73         callback.call(this, [true]);
 74     },
 75 
 76     /**
 77      * [abstract] This function should return the title of the job wizard step.
 78      */
 79     getTitle : function() {
 80         return "";
 81     },
 82 
 83     /**
 84      * Gets the text to show on the 'Previous' button
 85      */
 86     getPreviousText : function() {
 87         return '« Previous';
 88     },
 89 
 90     /**
 91      * Gets the Icon class to use for the 'Previous' button
 92      */
 93     getPreviousIconClass : function() {
 94         return undefined;
 95     },
 96 
 97     /**
 98      * Gets the text to show on the 'Next' button
 99      */
100     getNextText : function() {
101         return 'Next »';
102     },
103 
104     /**
105      * Gets the Icon class to use for the 'next' button
106      */
107     getNextIconClass : function() {
108         return undefined;
109     },
110 
111     /**
112      * Gets an array of portal.util.help.Instruction objects for providing help with this form item. If there
113      * are no relevant instructions, return null or an empty array.
114      */
115     getHelpInstructions : function() {
116         return null;
117     }
118 });
