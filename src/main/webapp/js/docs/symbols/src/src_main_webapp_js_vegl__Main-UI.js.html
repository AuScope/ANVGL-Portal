 17         var layersSorter = new Ext.util.Sorter({
 18             sorterFn: function(record1, record2) {
 19             	// 'order' is always received on the JSON
 20             	// if it is an empty string, the layers are sorted on layer name, ascending
 21                 var order1 = (record1.data.order.length ? record1.data.order : record1.data.name);
 22                 var order2 = (record2.data.order.length ? record2.data.order : record2.data.name);
 23                 return order1 > order2 ? 1 : (order1 < order2 ? -1 : 0);
 24             },
 25             direction: 'ASC'
 26         })
 27         
 28         var layersGrouper = new Ext.util.Grouper({
 29             groupFn: function(item) {
 30                 return item.data.group;
 31             },
 32             sorterFn: function(record1, record2) {
 33             	// 'order' is always received on the JSON
 34             	// if it is an empty string, the groups are sorted on group name, ascending
 35             	var order1 = (record1.data.order.length ? record1.data.order : record1.data.group);
 36             	var order2 = (record2.data.order.length ? record2.data.order : record2.data.group);
 37                 return order1 > order2 ? 1 : (order1 < order2 ? -1 : 0);
 38             },
 39             direction: 'ASC'
 40         });

 92         //Utility function for adding a new layer to the map
 93         //record must be a CSWRecord or KnownLayer
 94         var handleAddRecordToMap = function(sourceGrid, record) {
 95             if (!(record instanceof Array)) {
 96                 record = [record];
 97             }
 98 
 99             for( var z = 0; z < record.length; z++) {
100                 var newLayer = null;
101 
102                 //Ensure the layer DNE first
103                 var existingRecord = layerStore.getById(record[z].get('id'));
104                 if (existingRecord) {
105                     layersPanel.getSelectionModel().select([existingRecord], false);
106                     return;
107                  }
108 
109                 //Turn our KnownLayer/CSWRecord into an actual Layer
110                 if (record[z] instanceof portal.csw.CSWRecord) {
111                     newLayer = record[z].get('layer');
112                 } else {
113                     newLayer = record[z].get('layer');
114                 }
115 
116                 //if newLayer is undefined, it must have come from some other source like mastercatalogue
117                 if (!newLayer){
118                     newLayer = layerFactory.generateLayerFromCSWRecord(record[z])
119                     //we want it to display immediately.
120                     newLayer.set('displayed',true);
121                 }
122 
123                 //We may need to show a popup window with copyright info
124                 var cswRecords = newLayer.get('cswRecords');
125                 for (var i = 0; i < cswRecords.length; i++) {
126                     if (cswRecords[i].hasConstraints()) {
127                         var popup = Ext.create('portal.widgets.window.CSWRecordConstraintsWindow', {
128                             width : 625,
129                             cswRecords : cswRecords
130                         });
131 
132                         popup.show();
133 
134                         //HTML images may take a moment to load which stuffs up our layout
135                         //This is a horrible, horrible workaround.
136                         var task = new Ext.util.DelayedTask(function(){
137                             popup.doLayout();
138                         });
139                         task.delay(1000);
140 
141                         break;
142                     }
143                 }
144 
145                 layerStore.insert(0,newLayer); //this adds the layer to our store
146                 layersPanel.getSelectionModel().select([newLayer], false); //this ensures it gets selected
147             }
148         };
