1 /**
  2  * A window for allowing the user to review and accept terms and conditions
  3  *
  4  * {
  5  *  tccontent : String terms and conditions content
  6  * }
  7  *
  8  *  Adds the following events:
  9  *  {
 10  *      accept: function(win) - called if the user accepts the T&C's
 11  *      reject: function(win) - called if the user rejects the T&C's
 12  *  }
 13  *
 14  */
 15 Ext.define('vegl.widgets.TermsConditionsWindow', {
 16     extend : 'Ext.Window',
 17     alias: 'widget.tcwindow',
 18 
 19     constructor : function(config) {
 20         Ext.apply(config, {
 21             border: true,
 22             resizable: true,
 23             layout: 'fit',
 24             closable: false,
 25             modal: true,
 26             width: 600,
 27             height: 400,
 28             title: 'Terms and Conditions',
 29             items: [{
 30                 xtype: 'panel',
 31                 border: false,
 32                 padding: 5,
 33                 scrollable: true,
 34                 html: config.tccontent
 35             }],
 36             dockedItems: [{
 37                 xtype: 'toolbar',
 38                 dock: 'bottom',
 39                 items: [{
 40                     xtype: 'tbfill'
 41                 },{
 42                     xtype: 'button',
 43                     scale: 'large',
 44                     text: 'Reject',
 45                     handler: function(btn) {
 46                         var w = btn.up('tcwindow');
 47                         w.fireEvent('reject', w);
 48                         w.close();
 49                     }
 50                 },{
 51                     xtype: 'button',
 52                     cls: 'important-button',
 53                     scale: 'large',
 54                     text: 'Accept',
 55                     handler: function(btn) {
 56                         var w = btn.up('tcwindow');
 57                         w.fireEvent('accept', w);
 58                         w.close();
 59                     }
 60                 }]
 61             }]
 62         });
 63 
 64         this.callParent(arguments);
 65     }
 66 });
