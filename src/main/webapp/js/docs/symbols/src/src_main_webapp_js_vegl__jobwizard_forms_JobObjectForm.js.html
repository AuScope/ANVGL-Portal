252     onComputeSelect : function(combo, records) {
253         if (!records) {
254             this.imageStore.removeAll();
255             this.computeTypeStore.removeAll();
256             return;
257         }
258 
259         this.getComponent('image-combo').clearValue();
260         this.getComponent('resource-combo').clearValue();
261         this.imageStore.load({
262             params : {
263                 computeServiceId : records.get('id')
264             }
265         });
266     },
267 
268     onImageSelect : function(combo, records) {
269         if (!records) {
270             this.computeTypeStore.removeAll();
271             return;
272         }
273 
274         this.getComponent('resource-combo').clearValue();
275         var selectedComputeService = this.getComponent('computeServiceId').getSelection();
276         this.computeTypeStore.load({
277             params : {
278                 computeServiceId : selectedComputeService.get('id'),
279                 machineImageId : records.get('imageId')
280             }
281         });

288     getNumDownloadRequests : function() {
289         request = ((window.XMLHttpRequest) ? new XMLHttpRequest() : new ActiveXObject("Microsoft.XMLHTTP"));
290         request.open("GET", "getNumDownloadRequests.do", false); //<-- false makes it a synchonous request!

297     beginValidation : function(callback) {
298         var jobObjectFrm = this;
299         var wizardState = this.wizardState;
300 
301         var numDownloadReqs = this.getNumDownloadRequests();
302 
303 
304 
305         //Ensure we have entered all appropriate fields
306         if (!jobObjectFrm.getForm().isValid()) {
307             callback(false);
308             return;
309         }
310 
311         //Then save the job to the database before proceeding
312         var values = jobObjectFrm.getForm().getValues();
313         values.seriesId = jobObjectFrm.wizardState.seriesId;
314         Ext.Ajax.request({
315             url : 'updateOrCreateJob.do',
316             params : values,
317             callback : function(options, success, response) {
318                 if (!success) {
319                     portal.widgets.window.ErrorWindow.showText('Error saving details', 'There was an unexpected error when attempting to save the details on this form. Please try again in a few minutes.');
320                     callback(false);
321                     return;
322                 }
323 
324                 var responseObj = Ext.JSON.decode(response.responseText);
325                 if (!responseObj.success) {
326                     portal.widgets.window.ErrorWindow.showText('Error saving details', 'There was an unexpected error when attempting to save the details on this form.', responseObj.msg);
327                     callback(false);
328                     return;
329                 }
330 
331                 jobObjectFrm.wizardState.jobId = responseObj.data[0].id;
332                 // Store user selected toolbox into wizard state. That toolbox
333                 // will be used to select relevant script templates or examples.
334                 wizardState.toolbox = jobObjectFrm.getForm().findField("computeVmId").getRawValue();
335 
336                 // Store selected resource limits into wizard state. These values will be included
337                 // in template generation (to ensure valid numbers of CPU's are chosen etc)
338                 var computeTypeId = jobObjectFrm.getComponent('resource-combo').getValue();
339                 var computeType = jobObjectFrm.computeTypeStore.getById(computeTypeId);
340                 wizardState.ncpus = computeType.get('vcpus');
341                 wizardState.nrammb = computeType.get('ramMB');
342 
343                 if (!wizardState.skipConfirmPopup && numDownloadReqs === 0) {
344                     Ext.Msg.confirm('Confirm',
345                             'No data set has been captured. Do you want to continue?',
346                             function(button) {
347                                 if (button === 'yes') {
348                                     wizardState.skipConfirmPopup = true;
349                                     callback(true);
350                                     return;
351                                 } else {
352                                     callback(false);
353                                     return;
354                                 }
355                         });
356                 } else {
357                     callback(true);
358                     return;
359                 }
360 
361             }
362         });
363     },
364 
365     createSeries : function(wizardState) {
366 
367         Ext.Ajax.request({
368             url: 'secure/createSeries.do',
369             callback : function(options, success, response) {
370                 if (success) {
371                     var responseObj = Ext.JSON.decode(response.responseText);
372                     if (responseObj.success && Ext.isNumber(responseObj.data[0].id)) {
373                         wizardState.seriesId = responseObj.data[0].id;
374                         return;
375                     } else {
376                         errorMsg = responseObj.msg;
377                         errorInfo = responseObj.debugInfo;
378                     }
379                 } else {
380                     errorMsg = "There was an internal error saving your series.";
381                     errorInfo = "Please try again in a few minutes or report this error to cg_admin@csiro.au.";
382                 }
383 
384                 portal.widgets.window.ErrorWindow.showText('Create new series', errorMsg, errorInfo);
385 
386                 return;
387             }
388         });
