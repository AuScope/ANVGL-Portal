1 /**
  2  * Class for parsing a set of portal.csw.CSWRecord objects request/response
  3  * using the Querier interface
  4  * 
  5  * The resulting Querier includes online resources of the selected 
  6  * CSWRecord object and its immediate child CSWRecord object(s).
  7  */
  8 Ext.define('vegl.layer.querier.csw.CSWQuerier', {
  9     extend: 'portal.layer.querier.Querier',
 10 
 11     constructor: function(config){
 12         this.callParent(arguments);
 13     },
 14 
 15     /**
 16      * See parent class for definition
 17      */
 18     query : function(queryTarget, callback) {
 19         var cswRecord = queryTarget.get('cswRecord');
 20         if (!cswRecord) {
 21             callback(this, [], queryTarget);
 22             return;
 23         }
 24                 
 25         var keywordsString = "";
 26         var keywords = cswRecord.get('descriptiveKeywords');
 27         for (var i = 0; i < keywords.length; i++) {
 28             keywordsString += keywords[i];
 29             if (i < (keywords.length - 1)) {
 30                 keywordsString += ', ';
 31             }
 32         }
 33 
 34         var panel = Ext.create('portal.layer.querier.BaseComponent', {
 35             border : false,
 36             autoScroll : true,
 37             items : [{
 38                 layout : 'fit',
 39                 items : [{
 40                     xtype : 'fieldset',
 41                     items : [{
 42                         xtype : 'displayfield',
 43                         fieldLabel : 'Source',
 44                         value : Ext.util.Format.format('<a target="_blank" href="{0}">Link back to registry</a>', cswRecord.get('recordInfoUrl'))
 45                     },{
 46                         xtype : 'displayfield',
 47                         fieldLabel : 'Title',
 48                         anchor : '100%',
 49                         value : cswRecord.get('name')
 50                     }, {
 51                         xtype : 'textarea',
 52                         fieldLabel : 'Abstract',
 53                         anchor : '100%',
 54                         value : cswRecord.get('description'),
 55                         readOnly : true
 56                     },{
 57                         xtype : 'displayfield',
 58                         fieldLabel : 'Keywords',
 59                         anchor : '100%',
 60                         value : keywordsString
 61                     },{
 62                         xtype : 'displayfield',
 63                         fieldLabel : 'Contact Org',
 64                         anchor : '100%',
 65                         value : cswRecord.get('contactOrg')
 66                     },{
 67                         fieldLabel : 'Resources',
 68                         xtype : 'onlineresourcepanel',
 69                         cswRecords : cswRecord
 70                     }]
 71                 }]
 72             }]
 73         });
 74 
 75         callback(this, [panel], queryTarget);
 76     }
 77 });
