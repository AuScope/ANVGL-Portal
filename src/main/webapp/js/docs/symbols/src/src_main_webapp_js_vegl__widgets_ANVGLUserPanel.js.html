1 /**
  2  * A Ext.grid.Panel specialisation for rendering a single ANVGLUser object
  3  *
  4  *
  5  * Adds the following events
  6  *
  7  */
  8 Ext.define('vegl.widgets.ANVGLUserPanel', {
  9     extend : 'Ext.panel.Panel',
 10     alias : 'widget.userpanel',
 11 
 12     /**
 13      * Accepts the config for a Ext.grid.Panel along with the following additions:
 14      *
 15      * user: Instance of ANVGLUser to render
 16      */
 17     constructor : function(config) {
 18         Ext.apply(config, {
 19             layout: 'fit',
 20             items: [{
 21                 xtype: 'form',
 22                 border: false,
 23                 padding: '10 20 20 5',
 24                 layout: 'anchor',
 25                 items: [{
 26                     xtype: 'label',
 27                     itemId: 'fullName',
 28                     text: 'Loading...',
 29                     style: {
 30                         'font-size': '24px',
 31                     }
 32                 },{
 33                     xtype: 'textfield',
 34                     itemId: 'email',
 35                     readOnly: true,
 36                     fieldLabel: 'Email',
 37                     margin: '20 0 0 0',
 38                     anchor: '100%',
 39                     plugins: [{
 40                         ptype: 'fieldhelptext',
 41                         text: 'The email used that will be used to contact you (Modify this at accounts.google.com)'
 42                     }],
 43                     fieldStyle: {
 44                         color: '#888888',
 45                         'background-color': '#eeeeee'
 46                     }
 47                 },{
 48                     xtype: 'textfield',
 49                     itemId: 'arnExecution',
 50                     name: 'arnExecution',
 51                     fieldLabel: 'Compute ARN',
 52                     anchor: '100%',
 53                     allowBlank: false,
 54                     allowOnlyWhitespace: false,
 55                     plugins: [{
 56                         ptype: 'fieldhelptext',
 57                         text: 'The Amazon resource name describing the compute EC2 resource to be used for job execution'
 58                     }]
 59                 },{
 60                     xtype: 'textfield',
 61                     itemId: 'arnStorage',
 62                     name: 'arnStorage',
 63                     fieldLabel: 'Storage ARN',
 64                     anchor: '100%',
 65                     allowBlank: false,
 66                     allowOnlyWhitespace: false,
 67                     plugins: [{
 68                         ptype: 'fieldhelptext',
 69                         text: 'The Amazon resource name describing the storage S3 resource to be used for job artifacts'
 70                     }]
 71                 }]
 72             }],
 73             dockedItems: [{
 74                 xtype: 'toolbar',
 75                 dock: 'bottom',
 76                 items: [{
 77                     xtype: 'tbfill'
 78                 },{
 79                     xtype: 'label',
 80                     itemId: 'status',
 81                     style: {
 82                         'color': 'gray'
 83                     }
 84                 },{
 85                     xtype: 'button',
 86                     scale: 'large',
 87                     text: 'Download AWS Policy',
 88                     handler: function() {
 89                         Ext.MessageBox.alert('TODO', 'This functionality is yet to be developed...');
 90                     }
 91                 },{
 92                     xtype: 'button',
 93                     cls: 'important-button',
 94                     scale: 'large',
 95                     text: 'Save Changes',
 96                     handler: function(btn) {
 97                         var formPanel = btn.up('userpanel').down('form');
 98                         var statusLabel = btn.up('userpanel').down('#status');
 99                         if (!formPanel.isValid()) {
100                             return;
101                         }
102 
103                         statusLabel.setText('Saving your changes...');
104                         Ext.Ajax.request({
105                             url: 'secure/setUser.do',
106                             params: formPanel.getValues(),
107                             callback: function(options, success, response) {
108                                 if (!success) {
109                                     statusLabel.setText('');
110                                     Ext.MessageBox.alert('Error', 'There was an error saving your changes. Please try refreshing the page.');
111                                     return;
112                                 }
113 
114                                 var responseObj = Ext.JSON.decode(response.responseText);
115                                 if (!responseObj.success) {
116                                     statusLabel.setText('');
117                                     Ext.MessageBox.alert('Error', 'There was an error saving your changes. Please try refreshing the page.');
118                                     return;
119                                 }
120 
121                                 statusLabel.setText('Saved!');
122                             }
123                         });
124                     }
125                 }]
126             }],
127             listeners: {
128                 afterrender: function(userPanel) {
129                     if (config.user) {
130                         userPanel.setUser(config.user);
131                     }
132                 }
133             }
134         });
135 
136         this.callParent(arguments);
137     },
138 
139     setUser: function(user) {
140         this.down('#email').setValue(user.get('email'));
141         this.down('#arnExecution').setValue(user.get('arnExecution'));
142         this.down('#arnStorage').setValue(user.get('arnStorage'));
143         this.down('#fullName').setText('Hello ' + user.get('fullName'));
144 
145     }
146 });
