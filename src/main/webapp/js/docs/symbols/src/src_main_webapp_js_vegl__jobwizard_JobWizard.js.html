1 /**
  2  * A GUI widget for stringing together a collection BaseJobWizardForm objects into a Job Wizard for VL
  3  *
  4  * A JobWizard is a specialised Ext.Panel
  5  */
  6 
  7 Ext.define('vegl.jobwizard.JobWizard', {
  8     extend : 'Ext.Panel',
  9     alias : 'widget.jobwizard',
 10 
 11     /**
 12      * The internal state of the wizard (managed by internal forms)
 13      */
 14     wizardState : null,
 15 
 16     /**
 17      * Creates a new JobObjectForm form configured to write/read to the specified global state
 18      *
 19      * {
 20      *  wizardState : Object - [Optional] The initial state to apply to all forms
 21      *  forms : String[] - An array of class names of various vegl.jobwizard.forms.BaseJobWizardForm implementations
 22      * }
 23      */
 24     constructor: function(config) {
 25         var jobWizard = this;
 26         jobWizard.wizardState = config.wizardState ? config.wizardState : {};
 27 
 28         var baseJobWizardForms = [];
 29         for (var i = 0; i < config.forms.length; i++) {
 30             baseJobWizardForms.push(Ext.create(config.forms[i], jobWizard.wizardState));
 31         }
 32 
 33         //Handler for whenever a form can't load data
 34         var loadingError = function() {
 35             Ext.Msg.alert('Internal Error', 'An error has occured whilst loading data from the server. Please try again in a few minutes.');
 36         };
 37 
 38         //Function for handling internal step changes
 39         var gotoStep = function(newStep, requireValidation) {
 40             var layout = jobWizard.getLayout();
 41             var setNewStep = function(layout, step) {
 42                 //Deactivate our current form
 43                 layout.activeItem.items.get(0).fireEvent('jobWizardDeactive');
 44 
 45                 //Move to our next step (if it is a valid value)
 46                 if (step >= 0) {
 47                     layout.setActiveItem(step);
 48                     layout.activeItem.items.get(0).fireEvent('jobWizardActive');
 49                 }
 50             };
 51 
 52             if (requireValidation) {
 53                 var currentCard = layout.activeItem;
 54                 var currentForm = currentCard.items.get(0);
 55 
 56                 currentForm.beginValidation(function(success) {
 57                     if (success) {
 58                         setNewStep(layout, newStep);
 59                     }
 60                 });
 61             } else {
 62                 setNewStep(layout, newStep);
 63             }
 64         };
 65 
 66         //Wrap our forms in cards to be displayed
 67         var items = [];
 68         for (var i = 0; i < baseJobWizardForms.length; i++) {
 69             frm = baseJobWizardForms[i];
 70             frm.on('jobWizardLoadException', loadingError); //subscribe to all loading errors
 71 
 72             buttons = [];
 73 
 74             if (i !== 0) {
 75                 buttons.push({
 76                     text: frm.getPreviousText(),
 77                     iconCls : frm.getPreviousIconClass(),
 78                     handler: Ext.bind(gotoStep, this, [i - 1, false])
 79                 });
 80             }
 81 
 82             if (frm.additionalButtons) {
 83                 for (var j = 0; j < frm.additionalButtons.length; j++) {
 84                     buttons.push(frm.additionalButtons[j]);
 85                 }
 86             }
 87 
 88             //Always show our "next step" but set it to < 0 if there is no more cards
 89             //This way the last card can still be validated and submitted. (it can
 90             //decide what to do from there)
 91             var nextStep = i + 1;
 92             if (nextStep >= baseJobWizardForms.length) {
 93                 nextStep = -1;
 94             }
 95             buttons.push({
 96                 text: frm.getNextText(),
 97                 iconCls : frm.getNextIconClass(),
 98                 handler: Ext.bind(gotoStep, this, [nextStep, true])
 99             });
100 
101             items.push({
102                 title : 'Step ' + (i + 1) + ': ' + frm.getTitle(),
103                 defaults: { border: false },
104                 buttons : buttons,
105                 items : [frm]
106             });
107         }
108 
109         var startingJobIndex = 0;
110 
111         Ext.apply(config, {
112             layout: 'card',
113             activeItem: startingJobIndex,
114             defaults: { layout:'fit', buttonAlign: 'right' },
115             border: false,
116             style: {
117                 'background-color' : 'white'
118             },
119             items: items
120         });
121 
122         this.callParent(arguments);
123 
124         this.on('render', function() {
125             var activeItem = baseJobWizardForms[startingJobIndex];
126             activeItem.fireEvent('jobWizardActive');
127         });
128     }
129 });
