1 /**
  2  * A Ext.grid.Panel specialisation for rendering the series
  3  * available to the current user.
  4  *
  5  * Adds the following events
  6  * selectseries : function(vegl.widgets.SeriesPanel panel, vegl.models.Series selection) - fires whenever a new Series is selected
  7  * refreshDetailsPanel : function(vegl.widgets.SeriesPanel panel, vegl.models.Series series) - fires whenever a Series is successfully deleted
  8  * error : function(vegl.widgets.SereisPanel panel, String message) - fires whenever a comms error occurs
  9  */
 10 Ext.define('vegl.widgets.FolderPanel', {
 11     extend : 'Ext.tree.Panel',
 12     alias : 'widgets.seriesfolderpanel',
 13 
 14     cancelSeriesAction : null,
 15     deleteSeriesAction : null,
 16     contextMenu : null,
 17     seriesStore : null,
 18 
 19     constructor : function(config) {
 20         
 21       
 22         
 23          this.seriesStore =Ext.create('Ext.data.Store', {
 24             model : 'vegl.models.Series',
 25             proxy : {
 26                 type : 'ajax',
 27                 url : 'secure/querySeries.do',
 28                 reader : {
 29                     type : 'json',
 30                     rootProperty : 'data'
 31                 },
 32                 listeners : {
 33                     exception : function(proxy, response, operation) {
 34                         responseObj = Ext.JSON.decode(response.responseText);
 35                         errorMsg = responseObj.msg;
 36                         errorInfo = responseObj.debugInfo;
 37                         portal.widgets.window.ErrorWindow.showText('Error', errorMsg, errorInfo);
 38                     }
 39                 }
 40             },               
 41             autoLoad : true
 42         });
 43         
 44         
 45         var treeStore = this.getBaseTreeStore();
 46         
 47         this.seriesStore.on('load',function(store, records, successful, operation, node, eOpts){
 48             var folderRoot = treeStore.getRoot().firstChild
 49             folderRoot.set('text',records[0].get('user'))
 50             for(var i=0; i < records.length;i++){
 51                 folderRoot.insertChild(0, { id:records[i].get('id') ,text: records[i].get('name'), leaf: true })
 52             }
 53             folderRoot.expand(true);           
 54         })
 55 
 56 
 57         Ext.apply(config, {
 58             store : treeStore,
 59             rootVisible: false,
 60             buttons: [{
 61                 text: 'Remove Folder',
 62                 tooltip: 'Delete all jobs from this folder.',
 63                 handler: Ext.bind(this.deleteSeries, this),
 64                 cls: 'x-btn-text-icon',
 65                 iconCls: 'remove'      
 66               },{
 67                 text: 'Add Folder',
 68                 tooltip: 'Add a folder for catagorizing',
 69                 handler: Ext.bind(this.addFolderForm, this),
 70                 cls: 'x-btn-text-icon',
 71                 iconCls: 'add'      
 72               }]
 73         });   
 74 
 75         this.callParent(arguments);
 76 
 77         this.on('select', this.onSeriesSelection, this);      
 78         this.on('viewready', function(folderPanel,eOpts){            
 79             folderPanel.store.on('nodeexpand', function(store, records, successful, eOpts){                
 80                 folderPanel.getSelectionModel().select(store.lastChild);
 81             });            
 82             
 83         });
 84        
 85     },
 86     
 87     getBaseTreeStore : function(){
 88         return Ext.create('Ext.data.TreeStore', {
 89             root: {
 90                 expanded: true,
 91                 children: [                 
 92                     { text: "User Email", expanded: false },
 93                   
 94                 ]
 95             }
 96         });
 97     },
 98    
 99     
100     refresh : function(){
101         this.store.getRoot().firstChild.removeAll();
102         this.seriesStore.load();
103         this.fireEvent('refreshDetailsPanel');
104     },
105     
106     addFolderForm : function(){
107         var me = this;
108         var win = Ext.create('Ext.window.Window', {
109             title: 'Folder Name',
110             height: 220,
111             width: 400,
112             frameHeader : false,
113             layout: 'fit',
114             items: {
115                 xtype: 'form',                
116                 defaultType: 'textfield',
117                 items: [{
118                     xtype:'displayfield',
119                     margin : '5 5 15 5',                    
120                     value : 'After creating the folder, you may drag and drop jobs from the job panel onto the folder. Please wait until the job have complete provisioning before attempting.'
121                 },{
122                     fieldLabel: 'Folder Name',
123                     margin : '5 5 5 5',
124                     name: 'seriesName',
125                     allowBlank: false  ,
126                     width : '90%'
127                 },{
128                     fieldLabel: 'Description',
129                     name: 'seriesDesc',
130                     margin : '5 5 5 5',
131                     allowBlank: false,
132                     width : '90%'
133                 }],
134                 buttons  : [{
135                     text: 'Add',
136                     handler: function() {
137                         var form = this.up('form').getForm();                        
138                         me.createFolder(form.getFieldValues().seriesName,form.getFieldValues().seriesDesc);
139                         this.up('window').close();
140                     }
141                 }]
142             }
143         });
144         
145         win.show();
146     },
147     
148     createFolder : function(seriesName, seriesDesc) {        
149        var me = this;
150         Ext.Ajax.request({
151             url: 'secure/createFolder.do',
152             params: {
153                 'seriesName': seriesName,
154                 'seriesDescription': seriesDesc
155             },
156             callback : function(options, success, response) {
157                 if (success) {
158                     me.refresh();                   
159                   return;
160                 } else {
161                     errorMsg = "There was an internal error saving your series.";
162                     errorInfo = "Please try again in a few minutes or report this error to cg_admin@csiro.au.";
163                 }
164 
165                 portal.widgets.window.ErrorWindow.showText('Create new series', errorMsg, errorInfo);
166                
167                 return;
168             }
169         });
170         
171     },
172     
173     deleteSeries:function(){
174         var me = this;
175         if(this.getSelectionModel().getSelection()[0] && this.getSelectionModel().getSelection()[0].isLeaf()){
176             var seriesName = this.getSelectionModel().getSelection()[0].get('text')//VT: we only support single selection
177         }else{
178             Ext.Msg.alert('Warning', 'Please select a job folder first');
179         }
180         if(seriesName=='default'){
181             Ext.Msg.alert('Warning', 'We do not support the deletion of default folder. All new jobs are dropped in this folder');
182             return;
183         }
184         var seriesId = this.seriesStore.findRecord('name',seriesName).get('id')
185         
186         Ext.Msg.confirm('Delete Folder', 'Are you sure you want to delete this folder and it\'s jobs?', function(btn){
187             if (btn == 'yes'){
188                 Ext.getBody().mask('Deleting folder...');
189                 Ext.Ajax.request({
190                     url: 'secure/deleteSeriesJobs.do',
191                     params: {
192                         'seriesId': seriesId                        
193                     },
194                     callback : function(options, success, response) {
195                         Ext.getBody().unmask();
196                         if (success) {
197                             me.refresh();                           
198                           return;
199                         } else {
200                             errorMsg = "There was an internal error deleting your folder.";
201                             errorInfo = "Please try again in a few minutes or report this error to cg_admin@csiro.au.";
202                         }
203 
204                         portal.widgets.window.ErrorWindow.showText('Delete folder', errorMsg, errorInfo);
205                        
206                         return;
207                     }
208                 });
209             }
210         });
211     },
212 
213     onSeriesSelection : function(sm, series) {
214         if(series.isLeaf()){
215             var name = series.get('text');
216             var returnSeries = this.seriesStore.findRecord('name',name)
217             this.fireEvent('selectseries', this, returnSeries);
218         }
219         
220     }
221 });
