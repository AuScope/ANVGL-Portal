1 /**
  2  * A panel for rendering a set of CSWRecords that have been (sub)selected by a bounding box.
  3  */
  4 Ext.define('vegl.widgets.CodeEditorField', {
  5     extend : 'Ext.form.field.TextArea',
  6 
  7     alias : 'widget.codeeditor',
  8 
  9     editor : null,
 10     /**
 11      * Accepts the following:
 12      * {
 13      *  region : portal.util.BBox - The selected area (defaults to 0,0,0,0)
 14      *  cswRecords : portal.csw.CSWRecord[] - The records to display info for.
 15      * }
 16      */
 17     constructor : function(config) {
 18 
 19 
 20         //Build our configuration object
 21         Ext.apply(config, {
 22             listeners: {
 23                 render: function(obj,eopts){
 24                     var element = document.getElementById(obj.getInputId());
 25                     this.editor = CodeMirror.fromTextArea(element,{
 26                         lineNumbers: true,
 27                         mode: config.mode
 28                     });
 29 
 30                     if (config.value) {
 31                         obj.setValue(config.value);
 32                     }
 33                 }
 34             }
 35         });
 36 
 37         this.callParent(arguments);
 38 
 39 
 40     },
 41 
 42     setValue : function(script){
 43         if(this.editor){
 44             this.editor.setValue(script);
 45         }
 46     },
 47 
 48     getValue : function(){
 49         if(this.editor){
 50             return this.editor.getValue();
 51         }else{
 52             return null;
 53         }
 54 
 55     },
 56 
 57     somethingSelected : function(){
 58         return this.editor.somethingSelected();
 59     },
 60 
 61     getCursor : function(start){
 62         return this.editor.getCursor(start);
 63     },
 64 
 65 
 66     replaceRange : function(replacement,from,to,origin){
 67         return this.editor.replaceRange(replacement,from,to,origin);
 68     }
 69 
 70 
 71 });
