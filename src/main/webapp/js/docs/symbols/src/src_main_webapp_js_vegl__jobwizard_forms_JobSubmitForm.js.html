1 /**
  2  * Job wizard form for reviewing and then submitting a job for prcessing
  3  *
  4  * Author - Josh Vote
  5  */
  6 Ext.define('vegl.jobwizard.forms.JobSubmitForm', {
  7     extend : 'vegl.jobwizard.forms.JobUploadForm',
  8 
  9     /**
 10      * Creates a new JobObjectForm form configured to write/read to the specified global state
 11      */
 12     constructor: function(wizardState) {
 13         this.additionalButtons = [{
 14             text : 'Save Job',
 15             qtip : 'Save this job for later submission',
 16             iconCls : 'disk-icon',
 17             handler : Ext.bind(this.saveJob, this)
 18         }];
 19 
 20         this.callParent(arguments);
 21     },
 22 
 23     //Validation means we go and submit the job
 24     beginValidation : function(callback) {
 25         var jobSubmitFrm = this;                
 26         Ext.getBody().mask('Submitting Job...');
 27         
 28         Ext.Ajax.request({
 29             url : 'secure/submitJob.do',
 30             params : {
 31                 jobId : jobSubmitFrm.wizardState.jobId
 32             },
 33             timeout : 1000 * 60 * 5, //5 minutes defined in milli-seconds
 34             callback : function(options, success, response) {
 35                 Ext.getBody().unmask();
 36                 var errorMsg, errorInfo;
 37 
 38                 if (success) {
 39                     var responseObj = Ext.JSON.decode(response.responseText);
 40                     msg = responseObj.msg;
 41                     if (responseObj.success) {
 42                         jobSubmitFrm.noWindowUnloadWarning = true;
 43                         callback(true);
 44                         window.location = 'joblist.html';
 45                         return;
 46                     } else {
 47                         errorMsg = responseObj.msg;
 48                         errorInfo = responseObj.debugInfo;
 49                     }
 50                 } else {
 51                     errorMsg = "There was an error submitting your script for processing.";
 52                     errorInfo = "Please try again in a few minutes or report this error to cg_admin@csiro.au.";
 53                 }
 54 
 55                 //Create an error object and pass it to custom error window
 56                 var errorObj = {
 57                     title : 'Failure',
 58                     message : errorMsg,
 59                     info : errorInfo
 60                 };
 61 
 62                 var errorWin = Ext.create('portal.widgets.window.ErrorWindow', {
 63                     errorObj : errorObj
 64                 });
 65                 errorWin.show();
 66 
 67                 callback(false);
 68             }
 69         });
 70     },
 71 
 72     getTitle : function() {
 73         return "Review job before submission...";
 74     },
 75 
 76     getNextText : function() {
 77         return 'Submit Job';
 78     },
 79 
 80     getNextIconClass : function() {
 81         return 'submit-icon';
 82     },
 83 
 84     saveJob : function() {
 85         //This is a giant misdirection - the job was already saved way back in the Job Object form. This part is just
 86         //making the user feel more involved in the process
 87         this.noWindowUnloadWarning = true;
 88         Ext.Msg.alert('Job Saved', 'Your job has been saved for later submission. You can attempt submission later from the <a href="joblist.html">Monitor Jobs</a> page. It is now safe to close this window.');
 89     }
 90 });
