1 /**
  2  * A Ext.grid.Panel specialisation for rendering the Jobs
  3  * available to the current user.
  4  *
  5  * Adds the following events
  6  * selectjob : function(vegl.widgets.SeriesPanel panel, vegl.models.Job selection) - fires whenever a new Job is selected
  7  */
  8 Ext.define('vegl.widgets.JobInputFilesPanel', {
  9     extend : 'Ext.grid.Panel',
 10     alias : 'widget.jobinputfilespanel',
 11 
 12     currentJobId : null,
 13     downloadAction : null,
 14     deleteAction : null,
 15 
 16     constructor : function(config) {
 17         var jobFilesGrid = this;
 18 
 19         this.currentJobId = config.currentJob ? config.currentJob.get("id") : config.currentJobId;
 20 
 21         //Action for downloading a single file
 22         this.downloadAction = new Ext.Action({
 23             text: 'Download to your machine.',
 24             disabled: true,
 25             iconCls: 'disk-icon',
 26             scope : this,
 27             handler: function() {
 28                 var item = jobFilesGrid.getSelectionModel().getSelection()[0];
 29                 var source = item.get('source');
 30                 if (source instanceof vegl.models.FileRecord) {
 31                     var params = {
 32                         jobId : this.currentJobId,
 33                         filename : source.get('name')
 34                     };
 35 
 36                     portal.util.FileDownloader.downloadFile("downloadInputFile.do", params);
 37                 } else if (source instanceof vegl.models.Download) {
 38                     portal.util.FileDownloader.downloadFile(source.get('url'));
 39                 }
 40             }
 41         });
 42 
 43         //Action for deleting a single file
 44         this.deleteAction = new Ext.Action({
 45             text: 'Delete.',
 46             disabled: true,
 47             iconCls: 'cross-icon',
 48             scope : this,
 49             handler: function() {
 50                 var item = jobFilesGrid.getSelectionModel().getSelection()[0];
 51                 var source = item.get('source');
 52 
 53                 if (source instanceof vegl.models.FileRecord) {
 54                     Ext.Ajax.request({
 55                         url: 'deleteFiles.do',
 56                         callback: Ext.bind(this.updateFileStore, this),
 57                         params: {
 58                             'fileName': source.get('name'),
 59                             'jobId': this.currentJobId
 60                         }
 61                     });
 62                 } else if (source instanceof vegl.models.Download) {
 63                     Ext.Ajax.request({
 64                         url: 'deleteDownloads.do',
 65                         callback: Ext.bind(this.updateFileStore, this),
 66                         params: {
 67                             'downloadId': source.get('id'),
 68                             'jobId': this.currentJobId
 69                         }
 70                     });
 71                 }
 72             }
 73         });
 74 
 75         Ext.apply(config, {
 76             viewConfig : {
 77                 emptyText : '<p class="centeredlabel">This job doesn\'t have any input files or service downloads configured. You can add some by using the add button below or by selecting a dataset from the <a href="gmap.html">main page</a>.</p>'
 78             },
 79             features : [Ext.create('Ext.grid.feature.Grouping',{
 80                 groupHeaderTpl: '{name} ({[values.rows.length]} {[values.rows.length > 1 ? "Items" : "Item"]})'
 81             })],
 82             plugins : [{
 83                 ptype : 'inlinecontextmenu',
 84                 align : 'left',
 85                 actions: [this.downloadAction, this.deleteAction]
 86             }],
 87             store : Ext.create('Ext.data.Store', {
 88                 model : 'vegl.widgets.JobInputFilesPanel.Item',
 89                 groupField : 'group',
 90                 data : []
 91             }),
 92             columns: [{ header: 'Name', width: 200, sortable: true, dataIndex: 'name'},
 93                       { header: 'Location', width: 200, dataIndex: 'localPath'},
 94                       { header: 'Details', flex : 1, dataIndex: 'details'}]
 95         });
 96 
 97         this.callParent(arguments);
 98 
 99         this.on('selectionchange', this._onSelectionChange, this);
100         this.on('celldblclick', this._onDblClick, this);
101 
102     },
103 
104     /**
105      * Updates the store by making AJAX requests for the current job object
106      */
107     updateFileStore : function() {
108         if (!this.currentJobId) {
109             this.getStore().removeAll();
110             return;
111         }
112 
113         //Fire off the first request for already uploaded input files 
114         var loadMask = new Ext.LoadMask({
115             msg : 'Requesting input files...',
116             target : this,
117             removeMask : true            
118         });
119         loadMask.show();
120 
121         Ext.Ajax.request({
122             url : 'listJobFiles.do',
123             params : {
124                 jobId : this.currentJobId
125             },
126             scope : this,
127             callback : function(options, success, response) {
128                 if (!success) {
129                     loadMask.hide();
130                     return;
131                 }
132 
133                 var responseObj = Ext.JSON.decode(response.responseText);
134                 if (!responseObj.success || !responseObj.data) {
135                     loadMask.hide();
136                     return;
137                 }
138 
139                 var fileRecords = [];
140                 for (var i = 0; i < responseObj.data.length; i++) {
141                     fileRecords.push(Ext.create('vegl.models.FileRecord', responseObj.data[i]));
142                 }
143 
144                 //Fire off a second request for the downloads
145                 Ext.Ajax.request({
146                     url : 'getJobDownloads.do',
147                     params : {
148                         jobId : this.currentJobId
149                     },
150                     scope : this,
151                     fileRecords : fileRecords,
152                     callback : function(options, success, response) {
153                         loadMask.hide();
154                         if (!success) {
155                             return;
156                         }
157 
158                         var responseObj = Ext.JSON.decode(response.responseText);
159                         if (!responseObj.success || !responseObj.data) {
160                             return;
161                         }
162 
163                         var downloads = [];
164                         for (var i = 0; i < responseObj.data.length; i++) {
165                             downloads.push(Ext.create('vegl.models.Download', responseObj.data[i]));
166                         }
167 
168                         this._setStoreData(options.fileRecords, downloads);
169                     }
170                 });
171             }
172         });
173     },
174 
175     /**
176      * Sets the store data from a set of vegl.models.FileRecord and vegl.models.Download objects
177      */
178     _setStoreData : function(fileRecords, downloads) {
179         var ds = this.getStore();
180         var dsData = [];
181 
182         for (var i = 0; i < fileRecords.length; i++) {
183             var fr = fileRecords[i];
184             dsData.push(Ext.create('vegl.widgets.JobInputFilesPanel.Item', {
185                 id : 'fr-' + fr.get('name'),
186                 name : fr.get('name'),
187                 description : 'This file will be made available to the job upon startup. It will be put in the same working directory as the job script.',
188                 details : Ext.util.Format.fileSize(fr.get('size')),
189                 localPath : 'Local directory',
190                 source: fr,
191                 group : 'Your Uploaded Files'
192             }));
193         }
194 
195         for (var i = 0; i < downloads.length; i++) {
196             var dl = downloads[i];
197             var hostNameMatches = /.*:\/\/(.*?)\//g.exec(dl.get('url'));
198             var hostName = (hostNameMatches && hostNameMatches.length >= 2) ? hostNameMatches[1] : dl.get('url');
199 
200             dsData.push(Ext.create('vegl.widgets.JobInputFilesPanel.Item', {
201                 id : 'dl-' + dl.get('id'),
202                 name : dl.get('name'),
203                 description : dl.get('description'),
204                 details : Ext.util.Format.format('Service call to <a href="{1}" target="_blank">{0}</a>.', hostName, dl.get('url')),
205                 localPath : dl.get('localPath'),
206                 source: dl,
207                 group : 'Remote Web Service Downloads'
208             }));
209         }
210 
211         ds.loadData(dsData);
212     },
213 
214     _onDblClick : function(view, td, cellIndex, record, tr, rowIndex, e, eOpts) {
215         var sm = this.getSelectionModel();
216 
217         this.getSelectionModel().select([record], false);
218     },
219 
220     _onSelectionChange : function(sm) {
221         var totalSelections = this.getSelectionModel().getSelection().length;
222         if (totalSelections == 0) {
223             this.downloadAction.setDisabled(true);
224             this.deleteAction.setDisabled(true);
225         } else {
226             this.downloadAction.setDisabled(false);
227             this.deleteAction.setDisabled(false);
228         }
229     },
230 
231     /**
232      * Reloads this store with all the files for the specified job
233      */
234     listFilesForJob : function(job) {
235         this.currentJob = job;
236         this._updateStore();
237     }
238 });

240 /**
241  * Represents a generic model for containing Download or File objects.
242  */
243 Ext.define('vegl.widgets.JobInputFilesPanel.Item', {
244     extend: 'Ext.data.Model',
245 
246     fields: [
247         { name: 'id', type: 'string' }, //Unique ID for this download/file
248         { name: 'name', type: 'string' }, //short name of this download/file
249         { name: 'description', type: 'string' }, //longer descriptiion of this download
250         { name: 'details', type: 'string'}, //The remote URL or file size
251         { name: 'localPath', type: 'string'}, //Where the file will be made available
252         { name: 'source', type: 'auto'}, //Either a vegl.models.FileRecord or a vegl.models.Download object.
253         { name: 'group', type: 'string'} //How is this item grouped? Unique group name
254     ],
255 
256     idProperty : 'id'
257 });
