1 /**
  2  * A factory class for creating instances of portal.layer.downloader.Downloader
  3  */
  4 Ext.define('vegl.layer.VeglDownloaderFactory', {
  5 
  6     extend : 'portal.layer.downloader.DownloaderFactory',
  7 
  8 
  9     _generateDownloader : function(wfsResources, wmsResources, wcsResources) {
 10         if (wfsResources.length > 0) {
 11             return Ext.create('portal.layer.downloader.wfs.WFSDownloader', {map : this.map});
 12         }
 13 
 14         if (wcsResources.length > 0) {
 15             return Ext.create('portal.layer.downloader.coverage.WCSDownloader', {map : this.map});
 16         }
 17 
 18         if (wmsResources.length > 0) {
 19             return Ext.create('portal.layer.downloader.wms.WMSDownloader', {map : this.map});
 20         }
 21 
 22         //Not having a downloader isn't a problem.
 23         return null;
 24     },
 25 
 26     /**
 27      * See parent class for defn
 28      */
 29     buildFromKnownLayer : function(knownLayer) {
 30         var allOnlineResources = knownLayer.getAllOnlineResources();
 31 
 32         var wmsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WMS);
 33         var wfsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WFS);
 34         var wcsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WCS);
 35 
 36         return this._generateDownloader(wfsResources, wmsResources, wcsResources);
 37     },
 38 
 39     /**
 40      * See parent class for defn
 41      */
 42     buildFromCswRecord : function(cswRecord) {
 43         var allOnlineResources = cswRecord.get('onlineResources');
 44 
 45         var wmsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WMS);
 46         var wfsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WFS);
 47         var wcsResources = portal.csw.OnlineResource.getFilteredFromArray(allOnlineResources, portal.csw.OnlineResource.WCS);
 48 
 49         return this._generateDownloader(wfsResources, wmsResources, wcsResources);
 50     }
 51 });
