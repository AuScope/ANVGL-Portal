1 /**
  2  * A Ext.grid.Panel specialisation for rendering the logs (if any)
  3  * for a particular job.
  4  *
  5  */
  6 Ext.define('vegl.widgets.JobLogsPanel', {
  7     extend : 'portal.widgets.tab.ActivePreRenderTabPanel',
  8     alias : 'widget.joblogspanel',
  9 
 10     currentJob : null,
 11 
 12     constructor : function(config) {
 13 
 14         Ext.apply(config, {
 15             autoScroll : true,
 16             items : [],
 17             dockedItems: [{
 18                 xtype: 'toolbar',
 19                 dock: 'bottom',
 20                 items: [{
 21                     xtype: 'tbfill'
 22                 },{
 23                     xtype: 'button',
 24                     text: 'Refresh',
 25                     iconCls: 'refresh-icon',
 26                     scope: this,
 27                     handler: function() {
 28                         if (this.currentJob) {
 29                             this.listLogsForJob(this.currentJob);
 30                         }
 31                     }
 32                 }]
 33             }]
 34         });
 35 
 36         this.callParent(arguments);
 37 
 38         this.clearLogs(true);
 39     },
 40 
 41     /**
 42      * Reloads this store with all the jobs for the specified series
 43      */
 44     listLogsForJob : function(job) {
 45         var loadMaskElement = null;
 46         if (this.rendered) {
 47             loadMaskElement = this.getEl();
 48             loadMaskElement.mask('Loading logs...');
 49         }
 50 
 51 
 52         this.clearLogs();
 53         this.currentJob = job;
 54 
 55         Ext.Ajax.request({
 56             url : 'secure/getSectionedLogs.do',
 57             params : {
 58                 jobId : job.get('id')
 59             },
 60             scope : this,
 61             callback : function(options, success, response) {
 62                 if (loadMaskElement) {
 63                     loadMaskElement.unmask();
 64                 }
 65 
 66                 if (!success) {
 67                     this.addEmptyTab('Error communicating with the server.');
 68                     return;
 69                 }
 70 
 71                 var responseObj = Ext.JSON.decode(response.responseText);
 72                 if (!responseObj || !responseObj.success) {
 73                     this.addEmptyTab('Couldn\'t find any logs for the selected job: ' + responseObj.msg);
 74                     return;
 75                 }
 76 
 77                 var sections = responseObj.data[0];
 78                 var components = [];
 79 
 80                 for (var sectionName in sections) {
 81                     components.push({
 82                         title : sectionName,
 83                         itemId : sectionName,
 84                         autoDestroy : true,
 85                         scrollable  : true,
 86                         layout : 'fit',
 87                         items : [Ext.create('vegl.widgets.CodeEditorField', {
 88                             title : 'Script Source',
 89                             mode : 'text/plain',
 90                             showAutoIndent : false,
 91                             showLineNumbers : true,
 92                             readOnly : true,
 93                             listModes : [{text: "Plain text", mime: "text/plain"}],
 94                             value : sections[sectionName],
 95                             modes: [{
 96                                 mime:           ['text/plain'],
 97                                 dependencies:   []
 98                             }]
 99                         })]
100                     });
101                 }
102 
103                 //Add our components and select the python tab (if it exists, otherwise use the 'Full' tab)
104                 if (components.length > 0) {
105                     this.add(components);
106                     if (!this.setActiveTab('Python')) {
107                         this.setActiveTab('Full');
108                     }
109                 } else {
110                     this.clearLogs();
111                     this.addEmptyTab('The selected job hasn\'t recorded any logs yet.');
112                 }
113             }
114         });
115     },
116 
117     /**
118      * Removes logs from this panel. Optionally adds a replacement tab indicating this panel is empty
119      */
120     clearLogs : function(addEmptyTab) {
121         this.currentJob = null;
122         this.removeAll(true);
123         if (addEmptyTab) {
124             this.addEmptyTab('No job selected.');
125         }
126         this.doLayout();
127     },
128 
129     /**
130      * Adds an empty tab with the specified text to this panel
131      */
132     addEmptyTab : function(text) {
133         this.add({
134             autoDestroy : true,
135             html : Ext.util.Format.format('<p class="centeredlabel">{0}</p>', text)
136         });
137     }
138 });
