138     /**
139      * Loads user's contact/signature details into the form
140      * Closes the window that loaded this form on failure
141      */
142     loadFormData : function() {
143         this.getForm().load({
144             url : 'secure/getUserSignature.do',
145             waitMsg : 'Loading contact details...',
146             failure : function(frm, action) {
147                 responseObj = Ext.JSON.decode(action.response.responseText);
148                 errorMsg = responseObj.msg;
149                 errorInfo = responseObj.debugInfo;
150                 portal.widgets.window.ErrorWindow.showText('Failure', errorMsg, errorInfo);
151                 popupWin = Ext.getCmp('jobRegisterWin');
152                 popupWin.close();
153             },
154             success : function(frm, action) {
155                 responseObj = Ext.JSON.decode(action.response.responseText);
156                 if (responseObj.success) {
157                     frm.setValues(responseObj.data[0]);
158                 }
159             }
160         });
161     },
162 
163     /**
164      * Inserts the user defined job metadata record into GeoNetwork
165      * Closes the window that loaded this form on success
166      */
167     insertRecord : function(formPanel) {
168         var form = formPanel.getForm();
169         if (!form.isValid()) {
170             return;
171         }
172         
173         var selectedJob = this.job;
174 
175 
176         Ext.getBody().mask('Registering Job...');
177         
178         form.submit({
179             url: 'secure/insertRecord.do',
180             params: {
181                 jobId : this.jobId
182             },
183             success: function(form, action) {
184                 Ext.getBody().unmask();
185                 if (!action.result.success) {
186                     Ext.Msg.alert('Failure', 'Job registration failed. Please try again in a few minutes or report this error to cg_admin@csiro.au.');
187                     return;
188                 }
189                 responseObj = Ext.JSON.decode(action.response.responseText);
190                 //Set current selected job 'registeredUrl' attribute so that 
191                 //the user can see the url upon successfully registration.
192                 selectedJob.set('registeredUrl', responseObj.data);
193                 popupWin = Ext.getCmp('jobRegisterWin');
194                 popupWin.close();
195             },
196             failure: function(form, action) {
197                 Ext.getBody().unmask();
198                 responseObj = Ext.JSON.decode(action.response.responseText);
199                 errorMsg = responseObj.msg;
200                 errorInfo = responseObj.debugInfo;
201                 portal.widgets.window.ErrorWindow.showText('Failure', errorMsg, errorInfo);
202             }
203         });
204     }
