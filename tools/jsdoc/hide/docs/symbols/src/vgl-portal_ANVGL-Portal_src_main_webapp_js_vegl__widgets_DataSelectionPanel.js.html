1 /**
  2  * A panel for rendering a set of CSWRecords that have been (sub)selected by a bounding box.
  3  */
  4 Ext.define('vegl.widgets.DataSelectionPanel', {
  5     extend : 'Ext.grid.Panel',
  6 
  7     alias : 'widget.dataselectionpanel',
  8 
  9     /**
 10      * Accepts the following:
 11      * {
 12      *  region : portal.util.BBox - The selected area (defaults to 0,0,0,0)
 13      *  cswRecords : portal.csw.CSWRecord[] - The records to display info for.
 14      * }
 15      */
 16     constructor : function(config) {
 17         if (!config.region) {
 18             config.region = Ext.create('portal.util.BBox', {});
 19         }
 20         this.region = config.region;
 21 
 22         if (!config.cswRecords) {
 23             config.cswRecords = [];
 24         }
 25 
 26         var dataItems = vegl.widgets.DataSelectionPanelRow.parseCswRecords(config.cswRecords, this.region);
 27 
 28         var groupingFeature = Ext.create('Ext.grid.feature.Grouping',{
 29             groupHeaderTpl: '{name} ({[values.rows.length]} {[values.rows.length > 1 ? "Items" : "Item"]})'
 30         });
 31 
 32         var hideHeaders = true;
 33         if (typeof(config.hideHeaders) !== 'undefined' && config.hideHeaders != null) {
 34             hideHeaders = config.hideHeaders;
 35         }
 36 
 37         //Build our configuration object
 38         Ext.apply(config, {
 39             selModel : Ext.create('Ext.selection.CheckboxModel', {}),
 40             features : [groupingFeature],
 41             store : Ext.create('Ext.data.Store', {
 42                 groupField : 'resourceType',
 43                 model : 'vegl.widgets.DataSelectionPanelRow',
 44                 data : dataItems
 45             }),
 46             hideHeaders : hideHeaders,
 47             plugins : [{
 48                 ptype : 'selectablegrid'
 49             }],
 50             columns: [{
 51                 //Title column
 52                 dataIndex: 'name',
 53                 flex: 1,
 54                 renderer: Ext.bind(this._nameRenderer, this)
 55             },{
 56                 dataIndex: 'description',
 57                 width: 260,
 58                 renderer: Ext.bind(this._descriptionRenderer, this)
 59             },{
 60                 xtype: 'actioncolumn',
 61                 width:30,
 62                 sortable: false,
 63                 items: [{
 64                     icon: 'portal-core/img/download.png',
 65                     tooltip: 'Download this data to your local machine now',
 66                     scope : this,
 67                     handler: function(grid, rowIndex, colIndex, item, e, record) {
 68                         this._handleRowDownload(record);
 69                     }
 70                 }]
 71             },{
 72                 xtype: 'actioncolumn',
 73                 width:30,
 74                 sortable: false,
 75                 items: [{
 76                     icon: 'portal-core/img/edit.png',
 77                     tooltip: 'Edit the download options',
 78                     scope : this,
 79                     handler: function(grid, rowIndex, colIndex, item, e, record) {
 80                         this._handleRowEdit(record);
 81                     }
 82                 }]
 83             }],
 84             listeners: {
 85                 afterrender: function(panel) {
 86                     if (dataItems.length === 1) {
 87                         panel.setSelection(dataItems[0]);
 88                     }
 89                 }
 90             }
 91         });
 92 
 93         this.callParent(arguments);
 94 
 95         //Update the descriptions of all layers now they have been added
 96         for (var i = 0; i < dataItems.length; i++) {
 97             this._updateDescription(dataItems[i]);
 98         }
 99     },
100 
101     /**
102      * Handles the editing of a row by showing a popup and then updating the row description upon completion
103      */
104     _handleRowEdit : function(dataItem) {
105         var or = dataItem.get('onlineResource');
106         var dlOptions = dataItem.get('downloadOptions');
107 
108         vegl.util.DataSelectionUtil.showDownloadOptionsForResource(or, dlOptions, Ext.bind(function(updatedDlOptions) {
109             dataItem.set('downloadOptions', updatedDlOptions);
110             this._updateDescription(dataItem);
111         }, this));
112     },
113 
114     /**
115      * Handles the editing of a row by showing a popup and then updating the row description upon completion
116      */
117     _handleRowDownload : function(dataItem) {
118         var or = dataItem.get('onlineResource');
119         var dlOptions = dataItem.get('downloadOptions');
120 
121         vegl.util.DataSelectionUtil.makeDownloadUrl(or, dlOptions, false, function(success, dl) {
122             if (!success) {
123                 Ext.Msg.alert('Download Error', 'There was an error generating your download URL. Please try again in a few minutes or consider refreshing this page.');
124                 return;
125             }
126 
127             portal.util.FileDownloader.downloadFile(dl.get('url'),[],dlOptions.method);
128         });
129     },
130 
131     /**
132      * Saves every single selected row in this panel to the user's session. Errors are reported
133      * via a callback function
134      *
135      * @param callback function(totalSelected, totalErrors)
136      */
137     saveCurrentSelection: function(callback) {
138         var sm = this.getSelectionModel();
139         var selectedRows = sm.getSelection();
140 
141         if (selectedRows.length === 0) {
142             callback(0, 0);
143             return;
144         }
145 
146 
147         var totalResponses = 0;
148         var totalErrors = 0;
149 
150         var myMask = new Ext.LoadMask({
151             msg : 'Saving selected datasets...',
152             target : this
153         });
154 
155         myMask.show();
156 
157         //Simple handler for keeping track of how many requests we've sent vs how many responses we've received
158         var responseHandler = function(success) {
159             totalResponses++;
160             if (!success) {
161                 totalErrors++;
162             }
163 
164             if (totalResponses >= selectedRows.length) {
165                 myMask.hide();
166                 callback(totalResponses, totalErrors);
167             }
168         };
169 
170         //Different data sources have different functions to save
171         for (var i = 0; i < selectedRows.length; i++) {
172             vegl.util.DataSelectionUtil.makeDownloadUrl(selectedRows[i].get('onlineResource'),
173                     selectedRows[i].get('downloadOptions'),
174                     true,
175                     responseHandler);
176         }
177     },
178 
179     /**
180      * Updates the description text for the specified dataitem. The update may occur asynchronously
181      */
182     _updateDescription : function(dataItem) {
183         var or = dataItem.get('estimateResource');
184         var dlOptions = dataItem.get('downloadOptions');
185 
186         // WCS resources have their description based on the selected region (whatever that may be)
187         if (or.get('type') === portal.csw.OnlineResource.WCS) {
188             dataItem.set('description', 'Loading size details...');
189             vegl.util.WCSUtil.estimateCoverageSize(dlOptions, or.get('url'), dlOptions.layerName, dataItem, function(success, errorMsg, response, dataItem) {
190                 if (!success) {
191                     dataItem.set('description', errorMsg);
192                 } else {
193                     var approxTotal = Ext.util.Format.number(response.roundedTotal, '0,000');
194                     var approxSize = Ext.util.Format.fileSize(vegl.util.WCSUtil.roundToApproximation(response.width * response.height * 4));
195                     dataItem.set('description', Ext.util.Format.format('Approximately <b>{0}</b> data points in total.<br>Uncompressed that\'s roughly {1}', approxTotal, approxSize));
196                 }
197             });
198         } else if (or.get('type') === portal.csw.OnlineResource.WFS) {
199             dataItem.set('description', 'Loading size details...');
200             vegl.util.WFSUtil.estimateFeatureCount(dlOptions, or.get('url'), dlOptions.featureType, dataItem, function(success, errorMsg, response, dataItem) {
201                 if (!success) {
202                     dataItem.set('description', errorMsg);
203                 } else {
204                     var total = Ext.util.Format.number(response.total, '0,000');
205                     dataItem.set('description', Ext.util.Format.format('Approximately <b>{0}</b> features have been selected.', total));
206                 }
207             });
208         } else {
209             //Otherwise the description is pulled direct from the online resource
210             dataItem.set('description', or.get('description'));
211         }
212     },

214     _nameRenderer : function(value, metaData, record, row, col, store, gridView) {
215         var name = record.get('name');
216         var url = record.get('onlineResource').get('url');
217 
218         //Ensure there is a title (even it is just '<Untitled>'
219         if (!name || name.length === 0) {
220             name = '>Untitled<';
221         }
222 
223         //Render our HTML
224         return Ext.DomHelper.markup({
225             tag : 'div',
226             children : [{
227                 tag : 'b',
228                 html : name
229             },{
230                 tag : 'br'
231             },{
232                 tag : 'span',
233                 style : {
234                     color : '#555'
235                 },
236                 children : [{
237                     html : url
238                 }]
239             }]
240         });
241     },
242 
243     _descriptionRenderer : function(value, metaData, record, row, col, store, gridView) {
244         var type = record.get('onlineResource').get('type');
245         var description = record.get('description');
246 
247         //Render our HTML
248         switch(type) {
249         case portal.csw.OnlineResource.WWW:
250         case portal.csw.OnlineResource.FTP:
251         case portal.csw.OnlineResource.IRIS:
252         case portal.csw.OnlineResource.UNSUPPORTED:
253             return Ext.DomHelper.markup({
254                 tag : 'div',
255                 children : [{
256                     tag : 'a',
257                     target : '_blank',
258                     href : record.get('onlineResource').get('url'),
259                     children : [{
260                         tag : 'span',
261                         html : 'Click here to download.'
262                     }]
263                 }]
264             });
265         default:
266             return description;
267         }
268     }

271 Ext.define('vegl.widgets.DataSelectionPanelRow', {
272     extend : 'Ext.data.Model',
273 
274     statics : {
275         DATA_TYPE_PRIORITY: [portal.csw.OnlineResource.WCS, portal.csw.OnlineResource.NCSS, portal.csw.OnlineResource.WFS], //From low to high
276         ESTIMATE_TYPE_PRIORITY: [portal.csw.OnlineResource.WCS, portal.csw.OnlineResource.WFS], //From low to high
277 
278         /**
279          * Given an array of OnlineResources, choose the first item in that array with the highest priority.
280          * (will only return resources that match vegl.util.DataSelectionUtil.isResourceSupported)
281          */
282         getHighestPriorityType: function(onlineResources, priorityArray) {
283             //Gets the priority for a single online resource or -1.
284             var getPriority = function(or) {
285                 var currentPriority = -1;
286                 Ext.each(priorityArray, function(pType, index) {
287                     if (or.get('type') === pType ) {
288                         currentPriority = index;
289                         return false;
290                     }
291                 });
292                 return currentPriority;
293             };
294 
295             var highestPriority = -99;
296             var highestResource = null;
297             Ext.each(onlineResources, function(currentResource) {
298                 if (vegl.util.DataSelectionUtil.isResourceSupported(currentResource)) {
299                     var currentPriority = getPriority(currentResource);
300                     if (currentPriority > highestPriority) {
301                         highestPriority = currentPriority;
302                         highestResource = currentResource;
303                     }
304                 }
305             });
306 
307             return highestResource;
308         },
309 
310         /**
311          * Parses a single csw record and parent layer into an array of DataSelectionPanelRow items
312          */
313         parseCswRecord : function(cswRecord, defaultBbox) {
314             var dataItems = [];
315             var resources = cswRecord.get('onlineResources');
316 
317             var preferredDataServiceResource = vegl.widgets.DataSelectionPanelRow.getHighestPriorityType(resources, vegl.widgets.DataSelectionPanelRow.DATA_TYPE_PRIORITY);
318             var preferredEstimateServiceResource = vegl.widgets.DataSelectionPanelRow.getHighestPriorityType(resources, vegl.widgets.DataSelectionPanelRow.ESTIMATE_TYPE_PRIORITY);
319 
320             if (preferredDataServiceResource != null) {
321                 //Set the defaults of our new item
322                 var newItem = {
323                     resourceType : portal.csw.OnlineResource.typeToString(preferredDataServiceResource.get('type')),
324                     name : preferredDataServiceResource.get('name'),
325                     description : preferredDataServiceResource.get('description'),
326                     selected : true,
327                     cswRecord : cswRecord,
328                     onlineResource : preferredDataServiceResource,
329                     estimateResource : preferredEstimateServiceResource,
330                     downloadOptions : vegl.util.DataSelectionUtil.createDownloadOptionsForResource (preferredDataServiceResource, cswRecord, defaultBbox)
331                 };
332 
333                 dataItems.push(Ext.create('vegl.widgets.DataSelectionPanelRow', newItem));
334             }
335 
336             var childCswRecords = cswRecord.get('childRecords');
337             if (childCswRecords) {
338                 for (var i = 0; i < childCswRecords.length; i++) {
339                     dataItems = dataItems.concat(vegl.widgets.DataSelectionPanelRow.parseCswRecord(childCswRecords[i], defaultBbox));
340                 }
341             }
342 
343             return dataItems;
344         },
345 
346         /**
347          * Parses a CSWRecord array into an array of simple JS objects for usage with the internal data store for this panel
348          */
349         parseCswRecords : function(cswRecords, defaultBbox) {
350             var dataItems = [];
351             for (var i = 0; i < cswRecords.length; i++) {
352                 var cswRecord = cswRecords[i];
353 
354                 dataItems = dataItems.concat(vegl.widgets.DataSelectionPanelRow.parseCswRecord(cswRecord, defaultBbox));
355           }
356 
357           return dataItems;
358         }
