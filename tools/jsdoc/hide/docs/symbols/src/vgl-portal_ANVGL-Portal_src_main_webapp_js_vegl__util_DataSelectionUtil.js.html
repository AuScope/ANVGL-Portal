  8     /**
  9      * Given an OnlineResource, work out whether the type of that resource is supported
 10      * for data selection.
 11      *
 12      * @param or portal.csw.OnlineResource The resource to query
 13      */
 14     vegl.util.DataSelectionUtil.isResourceSupported = function(or) {
 15         switch(or.get('type')) {
 16         case portal.csw.OnlineResource.WCS:
 17         case portal.csw.OnlineResource.WFS:
 18         case portal.csw.OnlineResource.WWW:
 19         case portal.csw.OnlineResource.NCSS:
 20             return true;
 21 
 22         default:
 23             return false;
 24         }
 25     };

 27     /**
 28      * Creates a default download options object based on an online resource and CSWrecord.
 29      *
 30      * The object itself is used to maintain the state of dataset download options specific to
 31      * the type of the online resource. Its use is only intended within this class
 32      */
 33     vegl.util.DataSelectionUtil.createDownloadOptionsForResource = function(or, cswRecord, defaultBbox) {
 34         var dsBounds = cswRecord.get('geographicElements').length ? cswRecord.get('geographicElements')[0] : null;
 35 
 36         //Set the defaults of our new item
 37         var downloadOptions = {
 38             name : 'Subset of ' + or.get('name'),
 39             description : or.get('description'),
 40             url : or.get('url'),
 41             method : 'POST',
 42             localPath : '/tmp/' + or.get('name'),
 43             crs : (defaultBbox ? defaultBbox.crs : ''),
 44             eastBoundLongitude : (defaultBbox ? defaultBbox.eastBoundLongitude : 0),
 45             northBoundLatitude : (defaultBbox ? defaultBbox.northBoundLatitude : 0),
 46             southBoundLatitude : (defaultBbox ? defaultBbox.southBoundLatitude : 0),
 47             westBoundLongitude : (defaultBbox ? defaultBbox.westBoundLongitude : 0),
 48             dsEastBoundLongitude : (dsBounds ? dsBounds.eastBoundLongitude : null),
 49             dsNorthBoundLatitude : (dsBounds ? dsBounds.northBoundLatitude : null),
 50             dsSouthBoundLatitude : (dsBounds ? dsBounds.southBoundLatitude : null),
 51             dsWestBoundLongitude : (dsBounds ? dsBounds.westBoundLongitude : null)
 52         };
 53 
 54         //Add/subtract info based on resource type
 55         switch(or.get('type')) {
 56         case portal.csw.OnlineResource.WCS:
 57             delete downloadOptions.url;
 58             downloadOptions.format = 'nc';
 59             downloadOptions.layerName = or.get('name');
 60             downloadOptions.coverageName = downloadOptions.layerName;
 61             break;
 62         case portal.csw.OnlineResource.WFS:
 63 
 64             delete downloadOptions.url;
 65             downloadOptions.serviceUrl = or.get('url');
 66             downloadOptions.featureType = or.get('name');
 67             break;
 68         case portal.csw.OnlineResource.NCSS:
 69             downloadOptions.name = or.get('name');
 70             downloadOptions.method = 'GET';
 71             downloadOptions.layerName = or.get('name');
 72             downloadOptions.coverageName = downloadOptions.layerName;
 73             //delete eastBoundLongitude;
 74             //delete northBoundLatitude;
 75             //delete southBoundLatitude;
 76             //delete westBoundLongitude;
 77 
 78             //delete downloadOptions.url;
 79             //downloadOptions.serviceUrl = or.get('url');
 80             //downloadOptions.featureType = or.get('name');
 81             break;
 82         case portal.csw.OnlineResource.WWW:
 83             break;
 84 
 85         //We don't support EVERY type
 86         default:
 87             break;
 88         }
 89 
 90         return downloadOptions;
 91     };

263     /**
264      * Create a download request URL (in the form of a Download object).
265      * The request can optionally be stored into the user session
266      * for use later in the job submit phase of the workflow.
267      *
268      * @param or portal.csw.OnlineResource The resource to query
269      * @param dlOptions Object The current state of download options.
270      * @param saveInSession Boolean Should this download be saved into the users session for use during job submit?
271      * @param callback function(Boolean, vegl.models.Download) called whenever the save request returns. Success is passed as the first parameter
272      */
273     vegl.util.DataSelectionUtil.makeDownloadUrl = function(or, dlOptions, saveInSession, callback) {
274 
275         var ajaxResponseHandler = function(options, success, response, callback) {
276             if (!success) {
277                 callback(false, null);
278                 return;
279             }
280             var responseObj = Ext.JSON.decode(response.responseText);
281             if (!responseObj || !responseObj.success) {
282                 callback(false, null);
283                 return;
284             }
285 
286             callback(true, Ext.create('vegl.models.Download', responseObj.data));
287         };
288 
289         switch (or.get('type')) {
290         case portal.csw.OnlineResource.WCS:
291 
292             //Unfortunately ERDDAP requests that extend beyond the spatial bounds of the dataset
293             //will fail. To workaround this, we need to crop our selection to the dataset bounds
294             if (dlOptions.dsEastBoundLongitude != null && (dlOptions.dsEastBoundLongitude < dlOptions.eastBoundLongitude)) {
295                 dlOptions.eastBoundLongitude = dlOptions.dsEastBoundLongitude;
296             }
297             if (dlOptions.dsWestBoundLongitude != null && (dlOptions.dsWestBoundLongitude > dlOptions.westBoundLongitude)) {
298                 dlOptions.westBoundLongitude = dlOptions.dsWestBoundLongitude;
299             }
300             if (dlOptions.dsNorthBoundLatitude != null && (dlOptions.dsNorthBoundLatitude < dlOptions.northBoundLatitude)) {
301                 dlOptions.northBoundLatitude = dlOptions.dsNorthBoundLatitude;
302             }
303             if (dlOptions.dsSouthBoundLatitude != null && (dlOptions.dsSouthBoundLatitude > dlOptions.southBoundLatitude)) {
304                 dlOptions.southBoundLatitude = dlOptions.dsSouthBoundLatitude;
305             }
306 
307             Ext.Ajax.request({
308                 url : 'makeErddapUrl.do',
309                 params : Ext.apply({saveSession : saveInSession}, dlOptions),
310                 callback : Ext.bind(ajaxResponseHandler, this, [callback], true)
311             });
312             break;
313         case portal.csw.OnlineResource.WFS:
314             Ext.Ajax.request({
315                 url : 'makeWfsUrl.do',
316                 params : Ext.apply({saveSession : saveInSession}, dlOptions),
317                 callback : Ext.bind(ajaxResponseHandler, this, [callback], true)
318             });
319             break;
320         case portal.csw.OnlineResource.NCSS:
321             Ext.Ajax.request({
322                 url : 'makeNetcdfsubseserviceUrl.do',
323                 params : Ext.apply({saveSession : saveInSession}, dlOptions),
324                 callback : Ext.bind(ajaxResponseHandler, this, [callback], true)
325             });
326             break;
327 
328         default:
329             Ext.Ajax.request({
330                 url : 'makeDownloadUrl.do',
331                 params : Ext.apply({saveSession : saveInSession}, dlOptions),
332                 callback : Ext.bind(ajaxResponseHandler, this, [callback], true)
333             });
334             break;
335         }
336     };
