1 /**
  2  * A window for allowing the user to select the required MGA zone and enter the inversion cell dimensions
  3  *
  4  * {
  5  *  callback : function(WMSLayerFilterForm, int, int, int, int, int) Called when this window is closed and is passed the mgaZone and cellDimension values
  6  *  mgaZone : the MGA zone calculated based on the center of the selected area
  7  *  scope : the object to use as the scope
  8  * }
  9  *
 10  */
 11 Ext.define('vegl.widgets.InversionDimensionWindow', {
 12     extend : 'Ext.Window',
 13 
 14     constructor : function(config) {
 15 
 16         this.callback = config.callback;
 17         var parentWindow = this;
 18         var scope = config.scope;
 19 
 20         //subset file type selection values
 21         var mgaZoneStore = Ext.create('Ext.data.Store', {
 22             fields : ['type', 'value'],
 23             data : [{type : '49', value : 49},
 24                     {type : '50', value : 50},
 25                     {type : '51', value : 51},
 26                     {type : '52', value : 52},
 27                     {type : '53', value : 53},
 28                     {type : '54', value : 54},
 29                     {type : '55', value : 55},
 30                     {type : '56', value : 56}]
 31         });
 32 
 33         var mgaZoneCombo = Ext.create('Ext.form.ComboBox', {
 34             anchor         : '100%',
 35             editable       : false,
 36             forceSelection : true,
 37             fieldLabel     : 'MGA Zone',
 38             mode           : 'local',
 39             store          : mgaZoneStore,
 40             triggerAction  : 'all',
 41             typeAhead      : false,
 42             displayField   : 'type',
 43             valueField     : 'value',
 44             submitValue	   : false,
 45             value		   : config.mgaZone
 46         });
 47 
 48         var xCellDim = Ext.create('Ext.form.NumberField', {
 49             anchor      : '100%',
 50             fieldLabel  : 'X Cell Dimension',
 51             name        : 'xCellDim',
 52             value       : 0
 53         });
 54         var yCellDim = Ext.create('Ext.form.NumberField', {
 55             anchor      : '100%',
 56             fieldLabel  : 'Y Cell Dimension',
 57             name        : 'yCellDim',
 58             value       : 0
 59         });
 60         var zCellDim = Ext.create('Ext.form.NumberField', {
 61             anchor      : '100%',
 62             fieldLabel  : 'Z Cell Dimension',
 63             name        : 'zCellDim',
 64             value       : 0
 65         });
 66         var maxDepth = Ext.create('Ext.form.NumberField', {
 67             anchor      : '100%',
 68             fieldLabel  : 'Max Depth',
 69             name        : 'maxDepth',
 70             value       : 0
 71         });
 72 
 73         Ext.apply(config, {
 74             border          : true,
 75             resizable       : true,
 76             layout          : 'fit',
 77             modal           : true,
 78             plain           : false,
 79             buttonAlign     : 'right',
 80             title           : 'Enter Inversion Cell Dimensions',
 81             width           : 250,
 82             autoHeight 		: true,
 83             scope 			: this,
 84             items 		: [{
 85                 xtype : 'form',
 86                 items : [mgaZoneCombo,
 87                          xCellDim,
 88                          yCellDim,
 89                          zCellDim,
 90                          maxDepth]
 91             }],
 92             buttons:[{
 93                 xtype: 'button',
 94                 text: 'OK',
 95                 handler: function() {
 96 
 97                     if (xCellDim.getValue() > 0 || yCellDim.getValue() > 0 || zCellDim.getValue() > 0) {
 98                         parentWindow.callback(scope,
 99                                 mgaZoneCombo.getValue(),
100                                 xCellDim.getValue(),
101                                 yCellDim.getValue(),
102                                 zCellDim.getValue(),
103                                 maxDepth.getValue());
104                         parentWindow.close();
105                     } else {
106                         Ext.Msg.alert("Warning", "Cell dimensions must be greater than 0.");
107                     }
108                 }
109             },{
110                 xtype: 'button',
111                 text: 'Cancel',
112                 handler: function() {
113                     parentWindow.close();
114                 }
115             }]
116         });
117 
118         this.callParent(arguments);
119     }
120 });
