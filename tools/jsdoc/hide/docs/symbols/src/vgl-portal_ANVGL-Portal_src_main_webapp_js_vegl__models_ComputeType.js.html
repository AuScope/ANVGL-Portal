1 /**
  2  * A ComputeType is a unique ID of a Cloud Virtual Machine Compute Type/Flavor along with descriptive metadata
  3  */
  4 Ext.define('vegl.models.ComputeType', {
  5     extend: 'Ext.data.Model',
  6 
  7     fields: [
  8         { name: 'id', type: 'string' }, //Unique ID for this image
  9         { name: 'description', type: 'string' }, //Long description of this image
 10         { name: 'vcpus', type: 'string' }, //How many virtual CPUs
 11         { name: 'ramMB', type: 'string' }, //How much RAM (roughly) in MB does this  compute type offer
 12         { name: 'rootDiskGB', type: 'string' }, //How much does the root disk of this compute type offer (in GB)
 13         { name: 'ephemeralDiskGB', type: 'string' }, //How much does the Ephemeral disk of this compute type offer (in GB)
 14         { name: 'longDescription', type: 'string', convert: function(v, rec) {
 15                 var name = rec.get('description') ? rec.get('description') : rec.get('id');
 16                 return Ext.util.Format.format('{0} - {1} CPU(s), {2} MB RAM', name, rec.get('vcpus'), rec.get('ramMB'));
 17             }
 18         } //Long description of this compute type (calculated dynamically)
 19     ],
 20 
 21     idProperty : 'id'
 22 });
