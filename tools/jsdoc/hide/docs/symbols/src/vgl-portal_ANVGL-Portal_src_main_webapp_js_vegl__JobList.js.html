1 /**
  2  * Page for listing and monitoring running jobs
  3  */
  4 Ext.application({
  5     name : 'portal',
  6 
  7     //Here we build our GUI from existing components - this function should only be assembling the GUI
  8     //Any processing logic should be managed in dedicated classes - don't let this become a
  9     //monolithic 'do everything' function
 10     launch : function() {
 11         var onError = function(component, message) {
 12             Ext.Msg.show({
 13                 title: 'Error',
 14                 msg: message,
 15                 buttons: Ext.Msg.OK,
 16                 icon: Ext.Msg.ERROR
 17             });
 18         };
 19 
 20         var jobDetailsPanel = Ext.create('vegl.widgets.DetailsPanel', {
 21             title: 'Description',
 22             bodyStyle: 'padding:10px',
 23             listeners : {
 24                 error : onError
 25             }
 26         });
 27 
 28         var jobLogsPanel = Ext.create('vegl.widgets.JobLogsPanel', {
 29             title: 'Logs',
 30             bodyStyle: 'padding:10px',
 31             listeners : {
 32                 error : onError
 33             }
 34         });
 35 
 36         var jobFilesPanel = Ext.create('vegl.widgets.JobFilesPanel', {
 37             title: 'Files',
 38             stripeRows: true,
 39             listeners : {
 40                 error : onError
 41             }
 42         });
 43 
 44         var jobsPanel = Ext.create('vegl.widgets.JobsPanel', {
 45             title: 'Jobs of selected series',
 46             region: 'center',
 47             stripeRows: true,
 48             itemId : 'vgl-jobs-panel',
 49             listeners : {
 50                 selectjob : function(panel, job) {
 51                     jobDetailsPanel.showDetailsForJob(job);
 52                     jobFilesPanel.listFilesForJob(job);
 53                     jobLogsPanel.listLogsForJob(job);
 54                 },
 55                 refreshDetailsPanel : function(panel, series) {
 56                     jobDetailsPanel.showDetailsForSeries(series);
 57                     jobFilesPanel.cleanupDataStore();
 58                     jobLogsPanel.clearLogs(true);
 59                 },
 60                 refreshJobDescription : function(job) {
 61                     jobDetailsPanel.showDetailsForJob(job);
 62                 },
 63                 error : onError
 64             },
 65             viewConfig : {
 66                 listeners : {
 67                     contextmenu : function(view, index, node, e) {
 68                         if (!this.contextMenu) {
 69                             this.contextMenu = new Ext.menu.Menu({
 70                                 items: [ cancelSeriesAction, deleteSeriesAction ]
 71                             });
 72                         }
 73                         e.stopEvent();
 74                         this.contextMenu.showAt(e.getXY());
 75                     }
 76                 }
 77             }
 78         });
 79 
 80         var folderPanel = Ext.create('vegl.widgets.FolderPanel', {
 81             title: 'Folder Organization',
 82             region: 'north',
 83             itemId : 'vgl-series-panel',
 84             height: 250,
 85             split: true,
 86             stripeRows: true,
 87             listeners : {
 88                 selectseries : function(panel, series) {
 89                     jobsPanel.listJobsForSeries(series);
 90                     jobDetailsPanel.showDetailsForSeries(series);
 91                     jobFilesPanel.cleanupDataStore();
 92                     jobLogsPanel.clearLogs(true);
 93                 },
 94                 refreshDetailsPanel : function() {
 95                     jobsPanel.cleanupDataStore();
 96                     jobDetailsPanel.cleanupDetails();
 97                     jobLogsPanel.clearLogs(true);
 98                     jobFilesPanel.cleanupDataStore();
 99                 },
100                 error : onError
101             }
102         });
103 
104 
105         var me = this;
106 
107 
108         Ext.create('Ext.container.Viewport', {
109             layout: 'border',
110             id : 'vgl-joblist-viewport',
111             style: {
112                 'background-color': 'white'
113             },
114             listeners:{
115               afterrender : function(vp,eOpts){
116                   var GridDropTarget = new Ext.dd.DropTarget(folderPanel.getEl(), {
117                       ddGroup    : 'grid2tree',
118                       notifyDrop: function(dragsource, event, data) {
119                           var inprogressStatus=['Pending','Provisioning','In Queue'];
120                           if(Ext.Array.contains(inprogressStatus,data.records[0].get('status'))){
121                               Ext.Msg.alert('Warning', 'Please wait until the job has finished provisioning');
122                               return;
123                           }else{
124                               var folder = event.getTarget().textContent;
125                               me.updateJobSeries(data.records[0].get('id'),folder,function(){
126                                   jobsPanel.refreshJobsForSeries();
127                               });
128                           }
129 
130                       }
131 
132                   });
133               }
134             },
135             items: [{
136                 xtype: 'box',
137                 region: 'north',
138                 applyTo: 'body',
139                 height: 100
140             },{
141                 border: false,
142                 region: 'west',
143                 split: true,
144                 margins: '2 2 2 0',
145                 layout: 'border',
146                 width: 400,
147                 bodyStyle: {
148                     'background-color': 'white'
149                 },
150                 items: [folderPanel, jobsPanel]
151             },{
152                 xtype : 'tabpanel',
153                 title: 'Details',
154                 itemId : 'vgl-details-panel',
155                 region: 'center',
156                 margins: '2 2 2 0',
157                 activeTab: 0,
158                 split: true,
159                 items: [jobDetailsPanel, jobLogsPanel, jobFilesPanel]
160             }]
161         });
162     },
163 
164     updateJobSeries : function(jobId,newSeriesId,callback){
165         Ext.Ajax.request({
166             url: 'updateJobSeries.do',
167             params: {
168                 'id': jobId,
169                 'folderName': newSeriesId
170             },
171             callback : function(options, success, response) {
172                 if (success) {
173                     callback();
174                   return;
175                 } else {
176                     errorMsg = "There was an internal error saving your series.";
177                     errorInfo = "Please try again in a few minutes or report this error to cg_admin@csiro.au.";
178                 }
179 
180                 portal.widgets.window.ErrorWindow.showText('Create new series', errorMsg, errorInfo);
181 
182                 return;
183             }
184         });
185     }
186 });
