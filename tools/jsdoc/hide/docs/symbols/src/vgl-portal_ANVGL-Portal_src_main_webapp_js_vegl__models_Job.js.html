1 /**
  2  * A job is a collection of all input/processing/output information describing a single unit of work
  3  * within the VL workflow
  4  */
  5 Ext.define('vegl.models.Job', {
  6     extend: 'Ext.data.Model',
  7 
  8     statics : {
  9         STATUS_FAILED : "Failed",
 10         STATUS_ACTIVE : "Active",
 11         STATUS_PENDING : "Pending",
 12         STATUS_DONE : "Done",
 13         STATUS_CANCELLED : "Cancelled",
 14         STATUS_UNSUBMITTED : "Saved",
 15         STATUS_INQUEUE : "In Queue",
 16         STATUS_ERROR : "ERROR",
 17         STATUS_PROVISIONING : "Provisioning"
 18     },
 19 
 20     fields: [
 21         { name: 'id', type: 'int' }, //Unique identifier for the series
 22         { name: 'name', type: 'string' }, //Descriptive name of the series
 23         { name: 'description', type: 'string' }, //Long description of the series
 24         { name: 'emailAddress', type: 'string'}, //Email who created this series
 25         { name: 'user', type: 'string'}, //Username who created this series
 26         { name: 'submitDate', type: 'date', convert: function(value, record) {
 27             if (!value) {
 28                 return null;
 29             } else {
 30                 return new Date(value.time);
 31             }
 32         }}, //When this job was submitted to the cloud
 33         { name: 'processTimeLog', type: 'string'},
 34         { name: 'status', type: 'string'}, //The status of the job
 35 
 36         { name: 'computeVmId', type: 'string'}, //the ID of the VM that will be used to run this job
 37         { name: 'computeInstanceId', type: 'string'}, //the ID of the VM instance that is running this job (will be null if no job is currently running)
 38         { name: 'computeInstanceType', type: 'string'}, //The type of the compute instance to start (size of memory, number of CPUs etc) - eg m1.large. Can be null
 39         { name: 'computeInstanceKey', type: 'string'}, //The name of the key to inject into the instance at startup for root access. Can be null
 40         { name: 'computeServiceId', type: 'string'}, //The ID of the compute service used to execute this job
 41 
 42         { name: 'storageServiceId', type: 'string'}, //The ID of the storage service used to store this job's input/outputs
 43         { name: 'storageBaseKey', type: 'string'}, //The key prefix for all files associated with this job in the specified storage bucket
 44 
 45         { name: 'registeredUrl', type: 'string'},
 46         { name: 'seriesId', type: 'int'},
 47         { name: 'jobParameters', type: 'auto', convert: function(value, record) { //an array of vegl.models.Parameter objects
 48             if (!value) {
 49                 return [];
 50             } else {
 51                 var paramList = [];
 52                 for (var key in value) {
 53                     paramList.push(Ext.create('vegl.models.Parameter', value[key]));
 54                 }
 55                 return paramList;
 56             }
 57         }},
 58         { name: 'jobDownloads', type: 'auto', convert: function(value, record) { //an array of vegl.models.Download objects
 59             if (!value) {
 60                 return [];
 61             } else {
 62                 var paramList = [];
 63                 for (var i = 0; i < value.length; i++) {
 64                     if (value[i] instanceof vegl.models.Download) {
 65                         paramList.push(value[i]);
 66                     } else {
 67                         paramList.push(Ext.create('vegl.models.Download', value[i]));
 68                     }
 69                 }
 70                 return paramList;
 71             }
 72         }}
 73     ],
 74 
 75     /**
 76      * Gets the set of Parameters associated with this job encoded as a Key/Value encoded object.
 77      *
 78      * This object is readonly, changes made to it will NOT affect this Job object
 79      */
 80     getJobParametersObject : function() {
 81         var obj = {};
 82         var params = this.get('jobParameters');
 83         for (var i = 0; i < jobParameters.length; i++) {
 84             obj[params[i].get('name')] = params[i].get('value');
 85         }
 86 
 87         return obj;
 88     }
 89 });
