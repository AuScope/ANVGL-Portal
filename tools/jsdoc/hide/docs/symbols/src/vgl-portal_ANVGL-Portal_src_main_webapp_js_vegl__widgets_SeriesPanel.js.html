1 /**
  2  * A Ext.grid.Panel specialisation for rendering the series
  3  * available to the current user.
  4  *
  5  * Adds the following events
  6  * selectseries : function(vegl.widgets.SeriesPanel panel, vegl.models.Series selection) - fires whenever a new Series is selected
  7  * refreshDetailsPanel : function(vegl.widgets.SeriesPanel panel, vegl.models.Series series) - fires whenever a Series is successfully deleted
  8  * error : function(vegl.widgets.SereisPanel panel, String message) - fires whenever a comms error occurs
  9  */
 10 Ext.define('vegl.widgets.SeriesPanel', {
 11     extend : 'Ext.grid.Panel',
 12     alias : 'widgets.seriespanel',
 13 
 14     cancelSeriesAction : null,
 15     deleteSeriesAction : null,
 16     contextMenu : null,
 17 
 18     constructor : function(config) {
 19         
 20         this.cancelSeriesAction = new Ext.Action({
 21             text: 'Cancel series jobs',
 22             iconCls: 'cross-icon',
 23             scope : this,
 24             disabled : true,
 25             handler: function() {
 26                 var selection = this.getSelectionModel().getSelection();
 27                 if (selection.length > 0) {
 28                     this.cancelSeries(selection[0]);
 29                 }
 30             }
 31         });
 32 
 33         this.deleteSeriesAction = new Ext.Action({
 34             text: 'Delete entire series',
 35             iconCls: 'cross-icon',
 36             scope : this,
 37             disabled : true,
 38             handler: function() {
 39                 var selection = this.getSelectionModel().getSelection();
 40                 if (selection.length > 0) {
 41                     this.deleteSeries(selection[0]);
 42                 }
 43             }
 44         });
 45         
 46         var seriesStore =Ext.create('Ext.data.Store', {
 47             model : 'vegl.models.Series',
 48             proxy : {
 49                 type : 'ajax',
 50                 url : 'secure/querySeries.do',
 51                 reader : {
 52                     type : 'json',
 53                     rootProperty : 'data'
 54                 },
 55                 listeners : {
 56                     exception : function(proxy, response, operation) {
 57                         responseObj = Ext.JSON.decode(response.responseText);
 58                         errorMsg = responseObj.msg;
 59                         errorInfo = responseObj.debugInfo;
 60                         portal.widgets.window.ErrorWindow.showText('Error', errorMsg, errorInfo);
 61                     }
 62                 }
 63             },               
 64             autoLoad : true
 65         });
 66 
 67         Ext.apply(config, {
 68             plugins : [{
 69                 ptype : 'inlinecontextmenu',
 70                 align : 'center',
 71                 actions: [this.cancelSeriesAction, this.deleteSeriesAction]
 72             }],
 73             store : seriesStore,
 74             columns: [{ header: 'User', width: 150, sortable: true, dataIndex: 'user'},
 75                       { header: 'Series Name', flex : 1, sortable: true, dataIndex: 'name'}],
 76             buttons: [{
 77                 text: 'Query...',
 78                 tooltip: 'Displays the query dialog to search for job series',
 79                 handler: Ext.bind(this.onQuerySeries, this),
 80                 cls: 'x-btn-text-icon',
 81                 iconCls: 'find-icon'
 82             }]
 83         });   
 84 
 85         this.callParent(arguments);
 86 
 87         this.on('select', this.onSeriesSelection, this);
 88         this.on('selectionchange', this._onSelectionChange, this);   
 89         this.on('viewready', function(grid,eOpts){            
 90             grid.store.on('load', function(store, records, successful, eOpts){
 91                 grid.selModel.doSelect(store.data.items[0]);
 92             });            
 93             
 94         });
 95        
 96     },
 97 
 98     _onSelectionChange : function(sm) {
 99         var selections = this.getSelectionModel().getSelection();
100         if (selections.length === 0) {
101             this.cancelSeriesAction.setDisabled(true);
102             this.deleteSeriesAction.setDisabled(true);
103         } else {
104             this.cancelSeriesAction.setDisabled(false);
105             this.deleteSeriesAction.setDisabled(false);
106         }
107     },
108 
109     onSeriesSelection : function(sm, series) {
110         this.fireEvent('selectseries', this, series);
111     },
112 
113     onQuerySeries : function(btn) {
114         var me = this;
115 
116         var queryWindow = new Ext.Window({
117             title: 'Query job series',
118             plain: true,
119             width: 500,
120             resizable: false,
121             autoScroll: true,
122             constrainHeader: true,
123             bodyStyle: 'padding:5px;',
124             items: [{
125                 xtype : 'form',
126                 itemId : 'qForm',
127                 bodyStyle: 'padding:5px;',
128                 defaults: { anchor: "100%" },
129                 items: [{
130                     xtype: 'textfield',
131                     itemId: 'qSeriesName',
132                     fieldLabel: 'Series Name'
133                 }, {
134                     xtype: 'textfield',
135                     itemId: 'qSeriesDesc',
136                     fieldLabel: 'Description'
137                 }]
138             }],
139             modal: true,
140             buttons: [{
141                 text: 'Query',
142                 handler: function(btn) {
143                     var qWindow = btn.ownerCt.ownerCt;
144                     var qForm = qWindow.getComponent('qForm');
145                     var qName = qForm.getComponent('qSeriesName').getValue();
146                     var qDesc = qForm.getComponent('qSeriesDesc').getValue();
147                     me.querySeries(qName, qDesc);
148                     qWindow.close();
149                 }
150             }, {
151                 text: 'Cancel',
152                 handler: function(btn) {
153                     var qWindow = btn.ownerCt.ownerCt;
154                     qWindow.close();
155                 }
156             }]
157         });
158 
159         queryWindow.show();
160     },
161 
162     /**
163      * Update the contents of this panel with the specified query parameters
164      */
165     querySeries : function(name, desc) {
166         var store = this.getStore();
167         var ajaxProxy = store.getProxy();
168 
169         if (Ext.isEmpty(name)) {
170             ajaxProxy.extraParams.qSeriesName = null;
171         } else {
172             ajaxProxy.extraParams.qSeriesName = name;
173         }
174         if (Ext.isEmpty(desc)) {
175             ajaxProxy.extraParams.qSeriesDesc = null;
176         } else {
177             ajaxProxy.extraParams.qSeriesDesc = desc;
178         }
179         store.load();
180     },
181 
182     cancelSeries : function(series) {
183         Ext.Msg.show({
184             title: 'Cancel Series Jobs',
185             msg: 'Are you sure you want to cancel all jobs for the selected series?',
186             buttons: Ext.Msg.YESNO,
187             icon: Ext.Msg.WARNING,
188             modal: true,
189             closable: false,
190             scope : this,
191             fn: function(btn) {
192                 if (btn == 'yes') {
193 
194                     Ext.getBody().mask('Cancelling Series Jobs...');
195                     
196                     Ext.Ajax.request({
197                         url: 'secure/killSeriesJobs.do',
198                         params: { 'seriesId': series.get('id')},
199                         scope : this,
200                         callback : function(options, success, response) {                            
201                             Ext.getBody().unmask();
202                             if (!success) {
203                                 this.fireEvent('error', this, 'There was an error communicating with the VL server. Please try again later.');
204                                 return;
205                             }
206 
207                             var responseObj = Ext.JSON.decode(response.responseText);
208                             if (!responseObj.success) {
209                                 this.fireEvent('error', this, Ext.util.Format.format('There was an error cancelling the jobs for the selected series. {0}', responseObj.msg));
210                                 return;
211                             }
212 
213                             this.getStore().load();//refresh our store
214                         }
215                     });
216                 }
217             }
218         });
219     },
220 
221     deleteSeries : function(series) {
222         Ext.Msg.show({
223             title: 'Delete Series Jobs',
224             msg: 'Are you sure you want to delete this series and all jobs associated with it?',
225             buttons: Ext.Msg.YESNO,
226             icon: Ext.Msg.WARNING,
227             modal: true,
228             closable: false,
229             scope : this,
230             fn: function(btn) {
231                 if (btn == 'yes') {
232                     Ext.getBody().mask('Deleting Series...');
233                     Ext.Ajax.request({
234                         url: 'secure/deleteSeriesJobs.do',
235                         params: { 'seriesId': series.get('id')},
236                         scope : this,
237                         callback : function(options, success, response) {
238                             Ext.getBody().unmask();
239 
240                             if (!success) {
241                                 this.fireEvent('error', this, 'There was an error communicating with the VL server. Please try again later.');
242                                 return;
243                             }
244 
245                             var responseObj = Ext.JSON.decode(response.responseText);
246                             if (!responseObj.success) {
247                                 this.fireEvent('error', this, Ext.util.Format.format('There was an error deleting the jobs for the selected series. {0}', responseObj.msg));
248                                 return;
249                             }
250 
251                             this.getStore().load();//refresh our store
252                             this.fireEvent('refreshDetailsPanel'); //refresh Details panel
253                         }
254                     });
255                 }
256             }
257         });
258     }
259 });
