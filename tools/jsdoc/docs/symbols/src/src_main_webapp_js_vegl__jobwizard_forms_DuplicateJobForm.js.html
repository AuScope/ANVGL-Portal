1 /**
  2  * Job wizard form for handling duplication of an already run job.
  3  *
  4  * Author - Josh Vote
  5  */
  6 Ext.define('vegl.jobwizard.forms.DuplicateJobForm', {
  7     extend : 'vegl.jobwizard.forms.BaseJobWizardForm',
  8 
  9     cloudFilesStore : null,
 10     fileGrid : null,
 11 
 12     /**
 13      * Creates a new JobUploadForm form configured to write/read to the specified global state
 14      */
 15     constructor: function(wizardState) {
 16         var me = this;
 17 
 18         var downloadAction = new Ext.Action({
 19             text: 'Download',
 20             disabled: true,
 21             iconCls: 'disk-icon',
 22             handler: Ext.bind(this.downloadFile, this)
 23         });
 24 
 25         //Store for uploaded file details
 26         this.cloudFilesStore = Ext.create('Ext.data.Store', {
 27             model : 'vegl.models.FileRecord',
 28             autoLoad : true,
 29             filters : [
 30                 function(item) {
 31                     //These two files should not be copyable
 32                     return item.get('name') !== 'vl.sh.log' && item.get('name') !== 'workflow-version.txt';
 33                 }
 34             ],
 35             proxy : {
 36                 type : 'ajax',
 37                 url : 'secure/jobFiles.do',
 38                 extraParams : {
 39                     jobId : wizardState.jobId
 40                 },
 41                 reader : {
 42                     type : 'json',
 43                     rootProperty : 'data'
 44                 }
 45             }
 46         });
 47 
 48         this.fileGrid = Ext.create('Ext.grid.Panel', {
 49             title: 'Available files',
 50             store: this.cloudFilesStore,
 51             viewConfig : {
 52                 emptyText : '<p class="centeredlabel">This job doesn\'t have any input/output files to copy.</p>'
 53             },
 54             stripeRows: true,
 55             anchor: '100% -20',
 56             selModel : Ext.create('Ext.selection.CheckboxModel', {}),
 57             columns: [
 58                 { header: 'Filename', width: 200, sortable: true, dataIndex: 'name' },
 59                 { header: 'Size', width: 100, sortable: true, dataIndex: 'size',
 60                     renderer: Ext.util.Format.fileSize, align: 'right' }
 61             ]
 62         });
 63 
 64         this.callParent([{
 65             wizardState : wizardState,
 66             header : false,
 67             bodyStyle: 'padding:10px;',
 68             fileUpload: true,
 69             frame: true,
 70             listeners : {
 71                 jobWizardActive : Ext.bind(this.onActive, this)
 72             },
 73             items: [
 74                 this.fileGrid
 75             ]
 76         }]);
 77     },
 78 
 79     onActive : function() {
 80         //If we already have a jobId when this form activates, get rid of it.
 81         //When we move past this form we will be creating a new job with this ID
 82         if (this.wizardState.jobId) {
 83             this.deleteJobWithId(this.wizardState.jobId);
 84         }
 85 
 86         this.updateFileList();
 87     },
 88 
 89     deleteJobWithId : function(id) {      
 90         
 91         Ext.getBody().mask('Removing duplicate job...');
 92 
 93         //Tell the backend to remove duplicate job
 94         Ext.Ajax.request({
 95             url : 'secure/deleteJob.do',
 96             params : {
 97                 jobId : id
 98             },
 99             scope : this,
100             callback : function(options, success, response) {
101                 Ext.getBody().unmask();
102             }
103         });
104     },
105 
106     //Refresh the server side file list
107     updateFileList : function() {
108         var ajaxProxy = this.cloudFilesStore.getProxy();
109         ajaxProxy.extraParams.jobId = this.wizardState.duplicateJobId;
110         this.cloudFilesStore.load();
111     },
112 
113     //Validate by performing the job duplication
114     beginValidation : function(callback) {
115         //Don't duplicate more than once
116         if (this.performedDuplication) {
117             callback(true);
118             return;
119         }
120 
121         //Workout what files to duplicate
122         var filesToDuplicate = [];
123         var selectedFiles = this.fileGrid.getSelectionModel().getSelection();
124         for (var i = 0; i < selectedFiles.length; i++) {
125             filesToDuplicate.push(selectedFiles[i].get('name'));
126         }
127         var jobId = this.wizardState.duplicateJobId;        
128         
129         Ext.getBody().mask('Duplicating Job...');
130 
131         //Tell the backend to duplicate
132         Ext.Ajax.request({
133             url : 'secure/duplicateJob.do',
134             params : {
135                 jobId : jobId,
136                 file : filesToDuplicate
137             },
138             scope : this,
139             callback : function(options, success, response) {
140                 Ext.getBody().unmask();
141 
142                 if (success) {
143                     var responseObj = Ext.JSON.decode(response.responseText);
144                     if (responseObj.success && Ext.isNumber(responseObj.data[0].id)) {
145                         this.wizardState.jobId = responseObj.data[0].id;
146                         this.wizardState.seriesId = responseObj.data[0].seriesId;
147                         callback(true);
148                         return;
149                     }
150                 }
151 
152                 Ext.Msg.alert('Create new series','There was an internal error duplicating your job. Please try again in a few minutes.');
153                 callback(false);
154             }
155         });
156     },
157 
158     getTitle : function() {
159         return "Select files to copy into the new job...";
160     }
161 });
