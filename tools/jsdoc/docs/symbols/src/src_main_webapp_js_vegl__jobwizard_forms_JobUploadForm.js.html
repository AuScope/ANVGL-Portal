  4 Ext.define('vegl.jobwizard.forms.JobUploadForm', {
  5    /** 
  6      * @lends anvgl.JobSubmitForm.JobUploadForm 
  7      */ 
  8     extend : 'vegl.jobwizard.forms.BaseJobWizardForm',
  9     
 10     fileGrid : null,
 11     uploadedFilesStore : null,
 12     
 13     /**
 14      * Job wizard form for handling uploads of custom user input files. First interface of the 4-step job submission work-flow.
 15      * Creates a new JobUploadForm form configured to write/read to the specified global state
 16      *
 17      * @constructs
 18      * @param {object} wizardState
 19      */
 20     constructor: function(wizardState) {
 21         var jobUploadFrm = this;
 22         
 23         // execute the parent class
 24         this.callParent([{
 25             wizardState : wizardState,
 26             bodyStyle: 'padding:10px;',
 27             fileUpload: true,
 28             header : false,
 29             buttons: [],
 30             listeners : {
 31                     jobWizardActive : function() {
 32                         
 33                         if (this.wizardState.userAction == 'edit') {
 34                             jobUploadFrm.getForm().load({
 35                                 url : 'getJobObject.do',
 36                                 waitMsg : 'Loading Job Object...',
 37                                 params : {
 38                                     jobId : jobUploadFrm.wizardState.jobId
 39                                 },
 40                                 failure : Ext.bind(jobUploadFrm.fireEvent, jobUploadFrm, ['jobWizardLoadException']),
 41                                 success : function(frm, action) {
 42                                     var responseObj = Ext.JSON.decode(action.response.responseText);
 43 
 44                                     if (responseObj.success) {
 45                                         jobUploadFrm.wizardState.jobId = responseObj.data[0].id;
 46                                     }
 47                                 }
 48                             });
 49                             jobUploadFrm.updateFileList();
 50                         }
 51                         else {
 52                             // Ext.bind(jobUploadFrm.updateFileList, jobUploadFrm);
 53                             jobUploadFrm.updateFileList();
 54                         }
 55                     }
 56             },
 57             items: [{
 58                 xtype : 'jobinputfilespanel',
 59                 itemId : 'files-panel',
 60                 currentJobId : wizardState.jobId,
 61                 title: 'Input files',
 62                 stripeRows: true,
 63                 anchor: '100% -20',
 64                 buttons : [{
 65                     text : 'Add Input',
 66                     iconCls : 'add',
 67                     itemId : 'add-button',
 68                     align : 'right',
 69                     handler : function() {
 70                         Ext.create('vegl.widgets.JobInputFileWindow', {
 71                             jobId : jobUploadFrm.wizardState.jobId,
 72                             width : 500,
 73                             height : 300,
 74                             modal : true,
 75                             listeners : {
 76                                 close : function() {
 77                                     jobUploadFrm.updateFileList();
 78                                 }
 79                             }
 80                         }).show();
 81                     }
 82                 }]
 83             }]
 84         }]);
 85     },
 86 
 87     
 88     /**
 89      * Refresh the server side file list, and fetch all input-files associated with the job.
 90      * @function
 91      */
 92     updateFileList : function() {
 93         var filesPanel = this.getComponent('files-panel');
 94         filesPanel.currentJobId = this.wizardState.jobId;
 95         filesPanel.updateFileStore();
 96     },
 97 
 98     /**
 99      * 
100      * @function
101      */
102     getNumDownloadRequests : function() {
103         request = ((window.XMLHttpRequest) ? new XMLHttpRequest() : new ActiveXObject("Microsoft.XMLHTTP"));
104         // 'false' makes it a synchonous request!
105         request.open("GET", "getNumDownloadRequests.do", false); 
106         request.send(null);
107         
108         respObj = Ext.JSON.decode(request.responseText);
109         size = respObj.data;
110         
111         return size;
112     },
113     
114     
115     /**
116      * When the user wishes to proceed to the next step and clicks 'Next'
117      * @function
118      */
119     beginValidation : function(callback) {
120         callback(true);
121     },
122     
123     
124     /**
125      * Title of the interface
126      * @function
127      * @return {string} 
128      */
129     getTitle : function() {
130         return "Manage job input files...";
131     },
132 
133     
134     /**
135      * Gets the help instructions for the interface
136      * @function
137      * @return {object} instance of 'portal.util.help.Instruction'
138      */
139     getHelpInstructions : function() {
140         var filesPanel = this.getComponent('files-panel');
141         var addButton = filesPanel.queryById('add-button');
142         
143         // return the help instructions for the interface
144         return [Ext.create('portal.util.help.Instruction', {
145             highlightEl : filesPanel.getEl(),
146             title : 'Review Job Inputs',
147             anchor : 'top',
148             description : 'Every job has a number of input files. These files may yet to have been downloaded from a remote web service, or they may have already been uploaded from your desktop. Remote files will not be downloaded until after the job begins execution.<br/><br/>You can interact with these input files by right clicking them or by selecting them and pressing the \'Actions\' button.'
149         }), Ext.create('portal.util.help.Instruction', {
150             highlightEl : addButton.getEl(),
151             title : 'Add more Inputs',
152             anchor : 'top',
153             description : 'If you\'d like to add your own custom input files/web service downloads then press this button. You will be prompted to either upload a file from your local machine or to specify the public URL where the file can be accessed from.'
154         })];
155     }
156 });
