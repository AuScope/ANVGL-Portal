1 /**
  2  * AuScope implementation of the core portal FormFactory
  3  */
  4 Ext.define('vegl.layer.filterer.VeglFormFactory', {
  5     extend : 'portal.layer.filterer.FormFactory',
  6 
  7     /**
  8      * map : [Required] an instance of portal.map.BaseMap
  9      */
 10     constructor : function(config) {
 11         this.callParent(arguments);
 12     },
 13 
 14     /**
 15      * Given an portal.layer.Layer, work out whether there is an appropriate portal.layer.filterer.BaseFilterForm to show
 16      *
 17      * Returns a response in the form
 18      * {
 19      *    form : Ext.FormPanel - the formpanel to be displayed when this layer is selected (can be EmptyFilterForm)
 20      *    supportsFiltering : boolean - whether this formpanel supports the usage of the filter button
 21      *    layer : portal.layer.Layer that was used to generate this object
 22      * }
 23      *
 24      */
 25     getFilterForm : function(layer) {
 26         var baseFilterForm = null;
 27         var baseFilterFormCfg = {
 28             layer : layer,
 29             map : this.map
 30         };
 31 
 32         //A number of known layer's have specific filter forms
 33         if (layer.get('sourceType') === portal.layer.Layer.KNOWN_LAYER) {
 34             switch (layer.get('source').get('id')) {
 35                 case 'geophysics-datasets':
 36                     baseFilterForm = Ext.create('vegl.layer.filterer.forms.ProjectFilterForm', baseFilterFormCfg);
 37                     return this._generateResult(baseFilterForm, true);
 38             }
 39         }
 40 
 41         //otherwise let's see if we can guess an appropriate filter based on layer renderer
 42         if (layer.get('renderer') instanceof portal.layer.renderer.wms.LayerRenderer) {
 43             baseFilterForm = Ext.create('portal.layer.filterer.forms.WMSLayerFilterForm', baseFilterFormCfg);
 44             //VT: Filtering is support but for WMS, we want the image to be displayed immediately after it has been added and
 45             //the opacity can be adjusted from there on
 46             return this._generateResult(baseFilterForm, false);
 47         }
 48 
 49         //And otherwise we just show the empty filter form
 50         return this._generateResult(Ext.create('portal.layer.filterer.forms.EmptyFilterForm', baseFilterFormCfg), false);
 51     }
 52 });
