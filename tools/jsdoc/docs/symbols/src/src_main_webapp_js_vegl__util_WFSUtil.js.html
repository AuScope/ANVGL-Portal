1 /**
  2  * Utility functions for Web Feature Services
  3  */
  4 Ext.define('vegl.util.WFSUtil', {
  5     singleton: true
  6 }, function() {
  7     /**
  8      * Estimates the amount of data points the specified bbox will return from the specified WCS.
  9      *
 10      * Returns a JSON object via a callback with the following fields:
 11      *
 12      * {
 13      *  total : The total number of data features in the selection area
 14      *  roundedTotal : total rounded to a nice 'well rounded' number.
 15      * }
 16      *
 17      * @param bbox a portal.util.BBox
 18      * @param wfsUrl The wfs URL as a string
 19      * @param featureType the feature type name as a String
 20      * @param cbOpts Passed along to cb
 21      * @param cb function(success, errorMsg, responseObj, cbOpts) Called when estimation returns a result
 22      */
 23     vegl.util.WFSUtil.estimateFeatureCount = function(bbox, wfsUrl, featureType, cbOpts, cb) {
 24         Ext.Ajax.request({
 25             url : 'getFeatureCount.do',
 26             params : {
 27                 crs : bbox.crs,
 28                 northBoundLatitude : bbox.northBoundLatitude,
 29                 southBoundLatitude : bbox.southBoundLatitude,
 30                 eastBoundLongitude : bbox.eastBoundLongitude,
 31                 westBoundLongitude : bbox.westBoundLongitude,
 32                 serviceUrl : wfsUrl,
 33                 typeName : featureType
 34             },
 35             timeout : 1000 * 60 * 5, //5 minutes
 36             scope : this,
 37             callback : function(options, success, response) {
 38                 if (!success) {
 39                     cb(false, "Error estimating size: Couldn't contact VL server.", null, cbOpts);
 40                     return;
 41                 }
 42 
 43                 var responseObj = Ext.JSON.decode(response.responseText);
 44                 if (!responseObj.success) {
 45                     cb(false, 'Error estimating size: ' + responseObj.msg, null, cbOpts);
 46                     return;
 47                 }
 48 
 49                 cb(true, '', {
 50                     total : responseObj.data,
 51                     roundedTotal : vegl.util.WCSUtil.roundToApproximation(responseObj.data)
 52                 }, cbOpts);
 53             }
 54         });
 55     };
 56 });
