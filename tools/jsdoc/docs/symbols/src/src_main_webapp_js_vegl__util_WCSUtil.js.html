1 /**
  2  * Utility functions for Web Coverage Services
  3  */
  4 Ext.define('vegl.util.WCSUtil', {
  5     singleton: true
  6 }, function() {
  7     /**
  8      * Estimates the amount of data points the specified bbox will return from the specified WCS.
  9      *
 10      * Returns a JSON object via a callback with the following fields:
 11      *
 12      * {
 13      *  width : the returned number of data points horizontally (no guarantee on accuracy)
 14      *  height : the returned number of data points vertically (no guarantee on accuracy)
 15      *  roundedTotal : width * height rounded to a nice 'well rounded' number.
 16      * }
 17      *
 18      * @param bbox a portal.util.BBox
 19      * @param wcsUrl The wcs URL as a string
 20      * @param coverageName the coverage name as a String
 21      * @param cbOpts Passed along to cb
 22      * @param cb function(success, errorMsg, responseObj, cbOpts) Called when estimation returns a result
 23      */
 24     vegl.util.WCSUtil.estimateCoverageSize = function(bbox, wcsUrl, coverageName, cbOpts, cb) {
 25         Ext.Ajax.request({
 26             url : 'estimateCoverageSize.do',
 27             params : {
 28                 northBoundLatitude : bbox.northBoundLatitude,
 29                 southBoundLatitude : bbox.southBoundLatitude,
 30                 eastBoundLongitude : bbox.eastBoundLongitude,
 31                 westBoundLongitude : bbox.westBoundLongitude,
 32                 serviceUrl : wcsUrl,
 33                 coverageName : coverageName
 34             },
 35             scope : this,
 36             callback : function(options, success, response) {
 37                 if (!success) {
 38                     cb(false, 'Error estimating size: Couldn\'t contact VL server.', null, cbOpts);
 39                     return;
 40                 }
 41 
 42                 var responseObj = Ext.JSON.decode(response.responseText);
 43                 if (!responseObj.success) {
 44                     cb(false, 'Error estimating size: ' + responseObj.msg, null, cbOpts);
 45                     return;
 46                 }
 47 
 48                 cb(true, '', {
 49                     width : responseObj.data.width,
 50                     height : responseObj.data.height,
 51                     roundedTotal : vegl.util.WCSUtil.roundToApproximation(responseObj.data.width * responseObj.data.height)
 52                 }, cbOpts);
 53             }
 54         });
 55     };
 56 
 57     /**
 58      * Given a number with N digits, disregard most of the least significant digits and round
 59      * to a nice even number. This is to round off approximate values (eg 12456) to a more rounded
 60      * value so it looks more 'approximate' to people (eg 12000).
 61      */
 62     vegl.util.WCSUtil.roundToApproximation = function(number) {
 63         var totalDigits = number.toString().length;
 64         var digitsToKeep = Math.floor(totalDigits / 2);
 65         var divisor = Math.pow(10, totalDigits - digitsToKeep);
 66         return Math.round((number / divisor)) * divisor;
 67     };
 68 });
