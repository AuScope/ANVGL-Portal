1 /**
  2  * A Ext.panel.Panel specialisation for rendering details about a single job or series
  3  *
  4  */
  5 Ext.define('vegl.widgets.DetailsPanel', {
  6     extend : 'Ext.panel.Panel',
  7     alias : 'widgets.detailspanel',
  8 
  9     constructor : function(config) {
 10         Ext.apply(config, {
 11             items: [{
 12                 xtype: 'label',
 13                 itemId: 'title',
 14                 cls: 'jobdesc-title'
 15             },{
 16                 xtype: 'displayfield',
 17                 itemId: 'description',
 18                 fieldLabel: 'Description',
 19                 labelCls: 'jobdesc-key',
 20                 labelWidth: 120,
 21                 margin: '20 0 0 0'
 22             },{
 23                 xtype: 'displayfield',
 24                 itemId: 'submitDate',
 25                 labelCls: 'jobdesc-key',
 26                 labelWidth: 120,
 27                 fieldLabel: 'Submitted on',
 28             },{
 29                 xtype: 'displayfield',
 30                 itemId: 'registeredUrl',
 31                 labelCls: 'jobdesc-key',
 32                 labelWidth: 120,
 33                 fieldLabel: 'GeoNetwork url',
 34             }]
 35         });
 36 
 37         this.callParent(arguments);
 38     },
 39 
 40     /**
 41      * Updates the body of this panel with the contents of a vegl.models.Job
 42      */
 43     showDetailsForJob : function(job) {
 44         this.down('#title').setText(job.get('name'));
 45         this.down('#description').setValue(job.get('description'));
 46         this.down('#submitDate').setValue(job.get('submitDate'));
 47         this.down('#submitDate').setVisible(true);
 48         this.down('#registeredUrl').setValue(job.get('registeredUrl'));
 49         this.down('#registeredUrl').setVisible(true);
 50     },
 51 
 52     /**
 53      * Updates the body of this panel with the contents of a vegl.models.Series
 54      */
 55     showDetailsForSeries : function(series) {
 56         this.down('#title').setText(series.get('name'));
 57         this.down('#description').setValue(series.get('description'));
 58         this.down('#submitDate').setVisible(false);
 59         this.down('#submitDate').setValue('');
 60         this.down('#registeredUrl').setVisible(false);
 61         this.down('#registeredUrl').setValue('');
 62     },
 63 
 64     /**
 65      * Updates the body of this panel with an empty string
 66      */
 67     cleanupDetails : function() {
 68         this.down('#title').setText('');
 69         this.down('#description').setValue('');
 70         this.down('#submitDate').setValue('');
 71         this.down('#registeredUrl').setValue('');
 72     }
 73 });
