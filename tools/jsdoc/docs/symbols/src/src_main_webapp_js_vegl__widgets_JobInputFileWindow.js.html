1 /**
  2  * A window for creating a new job input file in the form of a vegl.models.Download or a vgl.models.FileRecord
  3  */
  4 Ext.define('vegl.widgets.JobInputFileWindow', {
  5     extend : 'Ext.Window',
  6 
  7     jobId : null,
  8 
  9     /**
 10      * Adds the following config options:
 11      *
 12      * {
 13      *  jobId : String - The unique ID of a job to add input files to
 14      * }
 15      */
 16     constructor : function(config) {
 17         this.jobId = config.jobId;
 18 
 19         Ext.apply(config, {
 20             layout : 'fit',
 21             title : 'New Job Input File',
 22             items : [{
 23                 xtype : 'tabpanel',
 24                 activeTab : 1,
 25                 itemId : 'tab-panel',
 26                 items : [{
 27                     xtype : 'form',
 28                     itemId : 'tab-local',
 29                     title : 'From Local File',
 30                     items : [{
 31                         xtype: 'filefield',
 32                         name: 'file',
 33                         anchor : '100%',
 34                         labelWidth: 150,
 35                         allowBlank: false,
 36                         fieldLabel: 'Select File to upload'
 37                     }]
 38                 },{
 39                     xtype : 'form',
 40                     itemId : 'tab-remote',
 41                     title : 'From Remote Service',
 42                     items : [{
 43                         xtype : 'textfield',
 44                         fieldLabel: 'URL',
 45                         anchor : '100%',
 46                         name : 'url',
 47                         emptyText : 'URL for the job to download.',
 48                         allowBlank : false
 49                     },{
 50                         xtype : 'textfield',
 51                         fieldLabel: 'Location',
 52                         anchor : '100%',
 53                         name : 'localPath',
 54                         emptyText : 'File path where the downloaded data will be stored.',
 55                         allowBlank : false
 56                     },{
 57                         xtype : 'textfield',
 58                         fieldLabel: 'Name',
 59                         anchor : '100%',
 60                         name : 'name',
 61                         allowBlank : false
 62                     },{
 63                         xtype : 'textarea',
 64                         fieldLabel: 'Description',
 65                         name : 'description',
 66                         anchor : '100%'
 67                     }]
 68                 }]
 69             }],
 70             buttons:[{
 71                 xtype: 'button',
 72                 text: 'Create Input',
 73                 scope : this,
 74                 iconCls : 'add',
 75                 handler: function() {
 76                     var tab = this.getComponent('tab-panel').getActiveTab();
 77                     if (tab.getItemId() === 'tab-local') {
 78                         this.addLocal(tab);
 79                     } else {
 80                         this.addRemote(tab);
 81                     }
 82                 }
 83             }]
 84         });
 85 
 86         this.callParent(arguments);
 87     },
 88 
 89     /**
 90      * Loads the file listed in the tab-local tab into the specified job.
 91      *
 92      * Closes this window on success
 93      */
 94     addLocal : function(formPanel) {
 95         var form = formPanel.getForm();
 96         if (!form.isValid()) {
 97             return;
 98         }
 99 
100         var params = form.getValues();
101         params.jobId = this.jobId;
102         
103         //Submit our form so our files get uploaded...
104         form.submit({
105             url: 'uploadFile.do',
106             params: params,
107             scope : this,
108             success: function(form, action) {
109                 if (!action.result.success) {
110                     Ext.Msg.alert('Error uploading file. ' + action.result.error);
111                     return;
112                 }
113                 this.close();
114             },
115             failure: function() {
116                 Ext.Msg.alert('Failure', 'File upload failed. Please try again in a few minutes.');
117             },
118             waitMsg: 'Uploading file, please wait...',
119             waitTitle: 'Upload file'
120         });
121     },
122 
123     /**
124      * Loads the remote input listed in the tab-remote tab into the specified job.
125      *
126      * Closes this window on success
127      */
128     addRemote : function(formPanel) {
129         var form = formPanel.getForm();
130         if (!form.isValid()) {
131             return;
132         }
133 
134 
135         var params = form.getValues();
136         params.id = this.jobId;
137         params.append = true;
138 
139         Ext.Ajax.request({
140             url: 'updateJobDownloads.do',
141             scope : this,
142             callback: function(options, success, response) {
143                 if (!success) {
144                     Ext.Msg.alert('Failure', 'Operation failed. Please try again in a few minutes.');
145                     return;
146                 }
147 
148                 var responseObj = Ext.JSON.decode(response.responseText);
149                 if (!responseObj.success) {
150                     Ext.create('portal.widgets.window.ErrorWindow', {errorObj : responseObj}).show();
151                     return;
152                 }
153 
154                 this.close();
155             },
156             params: params
157         });
158     }
159 });
